31 def _build_model(self, training=None):
32 model = tf.keras.models.Sequential()
33 norm = freezable_batch_norm.FreezableBatchNorm(training=training,
34 input_shape=(10,),
35 momentum=0.8)
36 model.add(norm)
37 return model, norm

39 def _train_freezable_batch_norm(self, training_mean, training_var):
40 model, _ = self._build_model()
41 model.compile(loss='mse', optimizer='sgd')
42
43 # centered on training_mean, variance training_var
44 train_data = np.random.normal(
45 loc=training_mean,
46 scale=training_var,
47 size=(1000, 10))
48 model.fit(train_data, train_data, epochs=4, verbose=0)
49 return model.weights

51 def _test_batchnorm_layer(
52 self, norm, should_be_training, test_data,
53 testing_mean, testing_var, training_arg, training_mean, training_var):
54 out_tensor = norm(tf.convert_to_tensor(test_data, dtype=tf.float32),
55 training=training_arg)
56 out = tf.keras.backend.eval(out_tensor)
57 out -= tf.keras.backend.eval(norm.beta)
58 out /= tf.keras.backend.eval(norm.gamma)
59
60 if not should_be_training:
61 out *= training_var
62 out += (training_mean - testing_mean)
63 out /= testing_var
64
65 np.testing.assert_allclose(out.mean(), 0.0, atol=1.5e-1)
66 np.testing.assert_allclose(out.std(), 1.0, atol=1.5e-1)
