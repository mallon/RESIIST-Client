47 def on_batch_end(self, batch, logs):
48 # grab the current learning rate and add log it to the list of
49 # learning rates that we've tried
50 lr = K.get_value(self.model.optimizer.lr)
51 self.lrs.append(lr)
52
53 # grab the loss at the end of this batch, increment the total
54 # number of batches processed, compute the average average
55 # loss, smooth it, and update the losses list with the
56 # smoothed value
57 l = logs["loss"]
58 self.batchNum += 1
59 self.avgLoss = (self.beta * self.avgLoss) + ((1 - self.beta) * l)
60 smooth = self.avgLoss / (1 - (self.beta ** self.batchNum))
61 self.losses.append(smooth)
62
63 # compute the maximum loss stopping factor value
64 stopLoss = self.stopFactor * self.bestLoss
65
66 # check to see whether the loss has grown too large
67 if self.batchNum > 1 and smooth > stopLoss:
68 # stop returning and return from the method
69 self.model.stop_training = True
70 return
71
72 # check to see if the best loss should be updated
73 if self.batchNum == 1 or smooth < self.bestLoss:
74 self.bestLoss = smooth
75
76 # increase the learning rate
77 lr *= self.lrMult
78 K.set_value(self.model.optimizer.lr, lr)

114 # learning rate, starting learning rate, and total number of
115 # batch updates
116 self.lrMult = (endLR / startLR) ** (1.0 / numBatchUpdates)

119 # then save the weights (so we can reset the weights when we
