16"""Base target assigner module.
17
18The job of a TargetAssigner is, for a given set of anchors (bounding boxes) and
19groundtruth detections (bounding boxes), to assign classification and regression
20targets to each anchor as well as weights to each anchor (specifying, e.g.,
21which anchors should not contribute to training loss).
22
23It assigns classification/regression targets by performing the following steps:
241) Computing pairwise similarity between anchors and groundtruth boxes using a
25 provided RegionSimilarity Calculator
262) Computing a matching based on the similarity matrix using a provided Matcher
273) Assigning regression targets based on the matching and a provided BoxCoder
284) Assigning classification targets based on the matching and groundtruth labels
29
30Note that TargetAssigners only operate on detections from a single
31image at a time, so any logic for applying a TargetAssigner to multiple
32images must be handled externally.
33"""

54class TargetAssigner(object):
55 """Target assigner to compute classification and regression targets."""

57 def __init__(self,
58 similarity_calc,
59 matcher,
60 box_coder,
61 negative_class_weight=1.0):
62 """Construct Object Detection Target Assigner.
63
64 Args:
65 similarity_calc: a RegionSimilarityCalculator
66 matcher: an object_detection.core.Matcher used to match groundtruth to
67 anchors.
68 box_coder: an object_detection.core.BoxCoder used to encode matching
69 groundtruth boxes with respect to anchors.
70 negative_class_weight: classification weight to be associated to negative
71 anchors (default: 1.0). The weight must be in [0., 1.].
72
73 Raises:
74 ValueError: if similarity_calc is not a RegionSimilarityCalculator or
75 if matcher is not a Matcher or if box_coder is not a BoxCoder
76 """
77 if not isinstance(similarity_calc, sim_calc.RegionSimilarityCalculator):
78 raise ValueError('similarity_calc must be a RegionSimilarityCalculator')
79 if not isinstance(matcher, mat.Matcher):
80 raise ValueError('matcher must be a Matcher')
81 if not isinstance(box_coder, bcoder.BoxCoder):
82 raise ValueError('box_coder must be a BoxCoder')
83 self._similarity_calc = similarity_calc
84 self._matcher = matcher
85 self._box_coder = box_coder
86 self._negative_class_weight = negative_class_weight

88 @property
89 def box_coder(self):
90 return self._box_coder

92 # TODO(rathodv): move labels, scores, and weights to groundtruth_boxes fields.
93 def assign(self,
94 anchors,
95 groundtruth_boxes,
96 groundtruth_labels=None,
97 unmatched_class_label=None,
98 groundtruth_weights=None):
99 """Assign classification and regression targets to each anchor.
100
101 For a given set of anchors and groundtruth detections, match anchors
102 to groundtruth_boxes and assign classification and regression targets to
103 each anchor as well as weights based on the resulting match (specifying,
104 e.g., which anchors should not contribute to training loss).
105
106 Anchors that are not matched to anything are given a classification target
107 of self._unmatched_cls_target which can be specified via the constructor.
108
109 Args:
110 anchors: a BoxList representing N anchors
111 groundtruth_boxes: a BoxList representing M groundtruth boxes
112 groundtruth_labels: a tensor of shape [M, d_1, ... d_k]
113 with labels for each of the ground_truth boxes. The subshape
114 [d_1, ... d_k] can be empty (corresponding to scalar inputs). When set
115 to None, groundtruth_labels assumes a binary problem where all
116 ground_truth boxes get a positive label (of 1).
117 unmatched_class_label: a float32 tensor with shape [d_1, d_2, ..., d_k]
118 which is consistent with the classification target for each
119 anchor (and can be empty for scalar targets). This shape must thus be
120 compatible with the groundtruth labels that are passed to the "assign"
121 function (which have shape [num_gt_boxes, d_1, d_2, ..., d_k]).
122 If set to None, unmatched_cls_target is set to be [0] for each anchor.
123 groundtruth_weights: a float tensor of shape [M] indicating the weight to
124 assign to all anchors match to a particular groundtruth box. The weights
125 must be in [0., 1.]. If None, all weights are set to 1. Generally no
126 groundtruth boxes with zero weight match to any anchors as matchers are
127 aware of groundtruth weights. Additionally, `cls_weights` and
128 `reg_weights` are calculated using groundtruth weights as an added
129 safety.
130
131 Returns:
132 cls_targets: a float32 tensor with shape [num_anchors, d_1, d_2 ... d_k],
133 where the subshape [d_1, ..., d_k] is compatible with groundtruth_labels
134 which has shape [num_gt_boxes, d_1, d_2, ... d_k].
135 cls_weights: a float32 tensor with shape [num_anchors, d_1, d_2 ... d_k],
136 representing weights for each element in cls_targets.
137 reg_targets: a float32 tensor with shape [num_anchors, box_code_dimension]
138 reg_weights: a float32 tensor with shape [num_anchors]
139 match: an int32 tensor of shape [num_anchors] containing result of anchor
140 groundtruth matching. Each position in the tensor indicates an anchor
141 and holds the following meaning:
142 (1) if match[i] >= 0, anchor i is matched with groundtruth match[i].
143 (2) if match[i]=-1, anchor i is marked to be background .
144 (3) if match[i]=-2, anchor i is ignored since it is not background and
145 does not have sufficient overlap to call it a foreground.
146
147 Raises:
148 ValueError: if anchors or groundtruth_boxes are not of type
149 box_list.BoxList
150 """
151 if not isinstance(anchors, box_list.BoxList):
152 raise ValueError('anchors must be an BoxList')
153 if not isinstance(groundtruth_boxes, box_list.BoxList):
154 raise ValueError('groundtruth_boxes must be an BoxList')
155
156 if unmatched_class_label is None:
157 unmatched_class_label = tf.constant([0], tf.float32)
158
159 if groundtruth_labels is None:
160 groundtruth_labels = tf.ones(tf.expand_dims(groundtruth_boxes.num_boxes(),
161 0))
162 groundtruth_labels = tf.expand_dims(groundtruth_labels, -1)
163
164 unmatched_shape_assert = shape_utils.assert_shape_equal(
165 shape_utils.combined_static_and_dynamic_shape(groundtruth_labels)[1:],
166 shape_utils.combined_static_and_dynamic_shape(unmatched_class_label))
167 labels_and_box_shapes_assert = shape_utils.assert_shape_equal(
168 shape_utils.combined_static_and_dynamic_shape(
169 groundtruth_labels)[:1],
170 shape_utils.combined_static_and_dynamic_shape(
171 groundtruth_boxes.get())[:1])
172
173 if groundtruth_weights is None:
174 num_gt_boxes = groundtruth_boxes.num_boxes_static()
175 if not num_gt_boxes:
176 num_gt_boxes = groundtruth_boxes.num_boxes()
177 groundtruth_weights = tf.ones([num_gt_boxes], dtype=tf.float32)
178
179 # set scores on the gt boxes
180 scores = 1 - groundtruth_labels[:, 0]
181 groundtruth_boxes.add_field(fields.BoxListFields.scores, scores)
182
183 with tf.control_dependencies(
184 [unmatched_shape_assert, labels_and_box_shapes_assert]):
185 match_quality_matrix = self._similarity_calc.compare(groundtruth_boxes,
186 anchors)
187 match = self._matcher.match(match_quality_matrix,
188 valid_rows=tf.greater(groundtruth_weights, 0))
189 reg_targets = self._create_regression_targets(anchors,
190 groundtruth_boxes,
191 match)
192 cls_targets = self._create_classification_targets(groundtruth_labels,
193 unmatched_class_label,
194 match)
195 reg_weights = self._create_regression_weights(match, groundtruth_weights)
196
197 cls_weights = self._create_classification_weights(match,
198 groundtruth_weights)
199 # convert cls_weights from per-anchor to per-class.
200 class_label_shape = tf.shape(cls_targets)[1:]
201 weights_shape = tf.shape(cls_weights)
202 weights_multiple = tf.concat(
203 [tf.ones_like(weights_shape), class_label_shape],
204 axis=0)
205 for _ in range(len(cls_targets.get_shape()[1:])):
206 cls_weights = tf.expand_dims(cls_weights, -1)
207 cls_weights = tf.tile(cls_weights, weights_multiple)
208
209 num_anchors = anchors.num_boxes_static()
210 if num_anchors is not None:
211 reg_targets = self._reset_target_shape(reg_targets, num_anchors)
212 cls_targets = self._reset_target_shape(cls_targets, num_anchors)
213 reg_weights = self._reset_target_shape(reg_weights, num_anchors)
214 cls_weights = self._reset_target_shape(cls_weights, num_anchors)
215
216 return (cls_targets, cls_weights, reg_targets, reg_weights,
217 match.match_results)

219 def _reset_target_shape(self, target, num_anchors):
220 """Sets the static shape of the target.
221
222 Args:
223 target: the target tensor. Its first dimension will be overwritten.
224 num_anchors: the number of anchors, which is used to override the target's
225 first dimension.
226
227 Returns:
228 A tensor with the shape info filled in.
229 """
230 target_shape = target.get_shape().as_list()
231 target_shape[0] = num_anchors
232 target.set_shape(target_shape)
233 return target

235 def _create_regression_targets(self, anchors, groundtruth_boxes, match):
236 """Returns a regression target for each anchor.
237
238 Args:
239 anchors: a BoxList representing N anchors
240 groundtruth_boxes: a BoxList representing M groundtruth_boxes
241 match: a matcher.Match object
242
243 Returns:
244 reg_targets: a float32 tensor with shape [N, box_code_dimension]
245 """
246 matched_gt_boxes = match.gather_based_on_match(
247 groundtruth_boxes.get(),
248 unmatched_value=tf.zeros(4),
249 ignored_value=tf.zeros(4))
250 matched_gt_boxlist = box_list.BoxList(matched_gt_boxes)
251 if groundtruth_boxes.has_field(fields.BoxListFields.keypoints):
252 groundtruth_keypoints = groundtruth_boxes.get_field(
253 fields.BoxListFields.keypoints)
254 matched_keypoints = match.gather_based_on_match(
255 groundtruth_keypoints,
256 unmatched_value=tf.zeros(groundtruth_keypoints.get_shape()[1:]),
257 ignored_value=tf.zeros(groundtruth_keypoints.get_shape()[1:]))
258 matched_gt_boxlist.add_field(fields.BoxListFields.keypoints,
259 matched_keypoints)
260 matched_reg_targets = self._box_coder.encode(matched_gt_boxlist, anchors)
261 match_results_shape = shape_utils.combined_static_and_dynamic_shape(
262 match.match_results)
263
264 # Zero out the unmatched and ignored regression targets.
265 unmatched_ignored_reg_targets = tf.tile(
266 self._default_regression_target(), [match_results_shape[0], 1])
267 matched_anchors_mask = match.matched_column_indicator()
268 reg_targets = tf.where(matched_anchors_mask,
269 matched_reg_targets,
270 unmatched_ignored_reg_targets)
271 return reg_targets
272
273 def _default_regression_target(self):
274 """Returns the default target for anchors to regress to.
275
276 Default regression targets are set to zero (though in
277 this implementation what these targets are set to should
278 not matter as the regression weight of any box set to
279 regress to the default target is zero).
280
281 Returns:
282 default_target: a float32 tensor with shape [1, box_code_dimension]
283 """
284 return tf.constant([self._box_coder.code_size*[0]], tf.float32)

286 def _create_classification_targets(self, groundtruth_labels,
287 unmatched_class_label, match):
288 """Create classification targets for each anchor.
289
290 Assign a classification target of for each anchor to the matching
291 groundtruth label that is provided by match. Anchors that are not matched
292 to anything are given the target self._unmatched_cls_target
293
294 Args:
295 groundtruth_labels: a tensor of shape [num_gt_boxes, d_1, ... d_k]
296 with labels for each of the ground_truth boxes. The subshape
297 [d_1, ... d_k] can be empty (corresponding to scalar labels).
298 unmatched_class_label: a float32 tensor with shape [d_1, d_2, ..., d_k]
299 which is consistent with the classification target for each
300 anchor (and can be empty for scalar targets). This shape must thus be
301 compatible with the groundtruth labels that are passed to the "assign"
302 function (which have shape [num_gt_boxes, d_1, d_2, ..., d_k]).
303 match: a matcher.Match object that provides a matching between anchors
304 and groundtruth boxes.
305
306 Returns:
307 a float32 tensor with shape [num_anchors, d_1, d_2 ... d_k], where the
308 subshape [d_1, ..., d_k] is compatible with groundtruth_labels which has
309 shape [num_gt_boxes, d_1, d_2, ... d_k].
310 """
311 return match.gather_based_on_match(
312 groundtruth_labels,
313 unmatched_value=unmatched_class_label,
314 ignored_value=unmatched_class_label)

316 def _create_regression_weights(self, match, groundtruth_weights):
317 """Set regression weight for each anchor.
318
319 Only positive anchors are set to contribute to the regression loss, so this
320 method returns a weight of 1 for every positive anchor and 0 for every
321 negative anchor.
322
323 Args:
324 match: a matcher.Match object that provides a matching between anchors
325 and groundtruth boxes.
326 groundtruth_weights: a float tensor of shape [M] indicating the weight to
327 assign to all anchors match to a particular groundtruth box.
328
329 Returns:
330 a float32 tensor with shape [num_anchors] representing regression weights.
331 """
332 return match.gather_based_on_match(
333 groundtruth_weights, ignored_value=0., unmatched_value=0.)
334
335 def _create_classification_weights(self,
336 match,
337 groundtruth_weights):
338 """Create classification weights for each anchor.
339
340 Positive (matched) anchors are associated with a weight of
341 positive_class_weight and negative (unmatched) anchors are associated with
342 a weight of negative_class_weight. When anchors are ignored, weights are set
343 to zero. By default, both positive/negative weights are set to 1.0,
344 but they can be adjusted to handle class imbalance (which is almost always
345 the case in object detection).
346
347 Args:
348 match: a matcher.Match object that provides a matching between anchors
349 and groundtruth boxes.
350 groundtruth_weights: a float tensor of shape [M] indicating the weight to
351 assign to all anchors match to a particular groundtruth box.
352
353 Returns:
354 a float32 tensor with shape [num_anchors] representing classification
355 weights.
356 """
357 return match.gather_based_on_match(
358 groundtruth_weights,
359 ignored_value=0.,
360 unmatched_value=self._negative_class_weight)

371# TODO(rathodv): This method pulls in all the implementation dependencies into
372# core. Therefore its best to have this factory method outside of core.
373def create_target_assigner(reference, stage=None,
374 negative_class_weight=1.0, use_matmul_gather=False):
375 """Factory function for creating standard target assigners.
376
377 Args:
378 reference: string referencing the type of TargetAssigner.
379 stage: string denoting stage: {proposal, detection}.
380 negative_class_weight: classification weight to be associated to negative
381 anchors (default: 1.0)
382 use_matmul_gather: whether to use matrix multiplication based gather which
383 are better suited for TPUs.
384
385 Returns:
386 TargetAssigner: desired target assigner.
387
388 Raises:
389 ValueError: if combination reference+stage is invalid.
390 """
391 if reference == 'Multibox' and stage == 'proposal':
392 similarity_calc = sim_calc.NegSqDistSimilarity()
393 matcher = bipartite_matcher.GreedyBipartiteMatcher()
394 box_coder = mean_stddev_box_coder.MeanStddevBoxCoder()
395
396 elif reference == 'FasterRCNN' and stage == 'proposal':
397 similarity_calc = sim_calc.IouSimilarity()
398 matcher = argmax_matcher.ArgMaxMatcher(matched_threshold=0.7,
399 unmatched_threshold=0.3,
400 force_match_for_each_row=True,
401 use_matmul_gather=use_matmul_gather)
402 box_coder = faster_rcnn_box_coder.FasterRcnnBoxCoder(
403 scale_factors=[10.0, 10.0, 5.0, 5.0])
404
405 elif reference == 'FasterRCNN' and stage == 'detection':
406 similarity_calc = sim_calc.IouSimilarity()
407 # Uses all proposals with IOU < 0.5 as candidate negatives.
408 matcher = argmax_matcher.ArgMaxMatcher(matched_threshold=0.5,
409 negatives_lower_than_unmatched=True,
410 use_matmul_gather=use_matmul_gather)
411 box_coder = faster_rcnn_box_coder.FasterRcnnBoxCoder(
412 scale_factors=[10.0, 10.0, 5.0, 5.0])
413
414 elif reference == 'FastRCNN':
415 similarity_calc = sim_calc.IouSimilarity()
416 matcher = argmax_matcher.ArgMaxMatcher(matched_threshold=0.5,
417 unmatched_threshold=0.1,
418 force_match_for_each_row=False,
419 negatives_lower_than_unmatched=False,
420 use_matmul_gather=use_matmul_gather)
421 box_coder = faster_rcnn_box_coder.FasterRcnnBoxCoder()
422
423 else:
424 raise ValueError('No valid combination of reference and stage.')
425
426 return TargetAssigner(similarity_calc, matcher, box_coder,
427 negative_class_weight=negative_class_weight)

430def batch_assign(target_assigner,
431 anchors_batch,
432 gt_box_batch,
433 gt_class_targets_batch,
434 unmatched_class_label=None,
435 gt_weights_batch=None):
436 """Batched assignment of classification and regression targets.
437
438 Args:
439 target_assigner: a target assigner.
440 anchors_batch: BoxList representing N box anchors or list of BoxList objects
441 with length batch_size representing anchor sets.
442 gt_box_batch: a list of BoxList objects with length batch_size
443 representing groundtruth boxes for each image in the batch
444 gt_class_targets_batch: a list of tensors with length batch_size, where
445 each tensor has shape [num_gt_boxes_i, classification_target_size] and
446 num_gt_boxes_i is the number of boxes in the ith boxlist of
447 gt_box_batch.
448 unmatched_class_label: a float32 tensor with shape [d_1, d_2, ..., d_k]
449 which is consistent with the classification target for each
450 anchor (and can be empty for scalar targets). This shape must thus be
451 compatible with the groundtruth labels that are passed to the "assign"
452 function (which have shape [num_gt_boxes, d_1, d_2, ..., d_k]).
453 gt_weights_batch: A list of 1-D tf.float32 tensors of shape
454 [num_boxes] containing weights for groundtruth boxes.
455
456 Returns:
457 batch_cls_targets: a tensor with shape [batch_size, num_anchors,
458 num_classes],
459 batch_cls_weights: a tensor with shape [batch_size, num_anchors,
460 num_classes],
461 batch_reg_targets: a tensor with shape [batch_size, num_anchors,
462 box_code_dimension]
463 batch_reg_weights: a tensor with shape [batch_size, num_anchors],
464 match: an int32 tensor of shape [batch_size, num_anchors] containing result
465 of anchor groundtruth matching. Each position in the tensor indicates an
466 anchor and holds the following meaning:
467 (1) if match[x, i] >= 0, anchor i is matched with groundtruth match[x, i].
468 (2) if match[x, i]=-1, anchor i is marked to be background .
469 (3) if match[x, i]=-2, anchor i is ignored since it is not background and
470 does not have sufficient overlap to call it a foreground.
471
472 Raises:
473 ValueError: if input list lengths are inconsistent, i.e.,
474 batch_size == len(gt_box_batch) == len(gt_class_targets_batch)
475 and batch_size == len(anchors_batch) unless anchors_batch is a single
476 BoxList.
477 """
478 if not isinstance(anchors_batch, list):
479 anchors_batch = len(gt_box_batch) * [anchors_batch]
480 if not all(
481 isinstance(anchors, box_list.BoxList) for anchors in anchors_batch):
482 raise ValueError('anchors_batch must be a BoxList or list of BoxLists.')
483 if not (len(anchors_batch)
484 == len(gt_box_batch)
485 == len(gt_class_targets_batch)):
486 raise ValueError('batch size incompatible with lengths of anchors_batch, '
487 'gt_box_batch and gt_class_targets_batch.')
488 cls_targets_list = []
489 cls_weights_list = []
490 reg_targets_list = []
491 reg_weights_list = []
492 match_list = []
493 if gt_weights_batch is None:
494 gt_weights_batch = [None] * len(gt_class_targets_batch)
495 for anchors, gt_boxes, gt_class_targets, gt_weights in zip(
496 anchors_batch, gt_box_batch, gt_class_targets_batch, gt_weights_batch):
497 (cls_targets, cls_weights,
498 reg_targets, reg_weights, match) = target_assigner.assign(
499 anchors, gt_boxes, gt_class_targets, unmatched_class_label, gt_weights)
500 cls_targets_list.append(cls_targets)
501 cls_weights_list.append(cls_weights)
502 reg_targets_list.append(reg_targets)
503 reg_weights_list.append(reg_weights)
504 match_list.append(match)
505 batch_cls_targets = tf.stack(cls_targets_list)
506 batch_cls_weights = tf.stack(cls_weights_list)
507 batch_reg_targets = tf.stack(reg_targets_list)
508 batch_reg_weights = tf.stack(reg_weights_list)
509 batch_match = tf.stack(match_list)
510 return (batch_cls_targets, batch_cls_weights, batch_reg_targets,
511 batch_reg_weights, batch_match)

518def batch_get_targets(batch_match, groundtruth_tensor_list,
519 groundtruth_weights_list, unmatched_value,
520 unmatched_weight):
521 """Returns targets based on anchor-groundtruth box matching results.
522
523 Args:
524 batch_match: An int32 tensor of shape [batch, num_anchors] containing the
525 result of target assignment returned by TargetAssigner.assign(..).
526 groundtruth_tensor_list: A list of groundtruth tensors of shape
527 [num_groundtruth, d_1, d_2, ..., d_k]. The tensors can be of any type.
528 groundtruth_weights_list: A list of weights, one per groundtruth tensor, of
529 shape [num_groundtruth].
530 unmatched_value: A tensor of shape [d_1, d_2, ..., d_k] of the same type as
531 groundtruth tensor containing target value for anchors that remain
532 unmatched.
533 unmatched_weight: Scalar weight to assign to anchors that remain unmatched.
534
535 Returns:
536 targets: A tensor of shape [batch, num_anchors, d_1, d_2, ..., d_k]
537 containing targets for anchors.
538 weights: A float tensor of shape [batch, num_anchors] containing the weights
539 to assign to each target.
540 """
541 match_list = tf.unstack(batch_match)
542 targets_list = []
543 weights_list = []
544 for match_tensor, groundtruth_tensor, groundtruth_weight in zip(
545 match_list, groundtruth_tensor_list, groundtruth_weights_list):
546 match_object = mat.Match(match_tensor)
547 targets = match_object.gather_based_on_match(
548 groundtruth_tensor,
549 unmatched_value=unmatched_value,
550 ignored_value=unmatched_value)
551 targets_list.append(targets)
552 weights = match_object.gather_based_on_match(
553 groundtruth_weight,
554 unmatched_value=unmatched_weight,
555 ignored_value=tf.zeros_like(unmatched_weight))
556 weights_list.append(weights)
557 return tf.stack(targets_list), tf.stack(weights_list)

560def batch_assign_confidences(target_assigner,
561 anchors_batch,
562 gt_box_batch,
563 gt_class_confidences_batch,
564 gt_weights_batch=None,
565 unmatched_class_label=None,
566 include_background_class=True,
567 implicit_class_weight=1.0):
568 """Batched assignment of classification and regression targets.
569
570 This differences between batch_assign_confidences and batch_assign_targets:
571 - 'batch_assign_targets' supports scalar (agnostic), vector (multiclass) and
572 tensor (high-dimensional) targets. 'batch_assign_confidences' only support
573 scalar (agnostic) and vector (multiclass) targets.
574 - 'batch_assign_targets' assumes the input class tensor using the binary
575 one/K-hot encoding. 'batch_assign_confidences' takes the class confidence
576 scores as the input, where 1 means positive classes, 0 means implicit
577 negative classes, and -1 means explicit negative classes.
578 - 'batch_assign_confidences' assigns the targets in the similar way as
579 'batch_assign_targets' except that it gives different weights for implicit
580 and explicit classes. This allows user to control the negative gradients
581 pushed differently for implicit and explicit examples during the training.
582
583 Args:
584 target_assigner: a target assigner.
585 anchors_batch: BoxList representing N box anchors or list of BoxList objects
586 with length batch_size representing anchor sets.
587 gt_box_batch: a list of BoxList objects with length batch_size
588 representing groundtruth boxes for each image in the batch
589 gt_class_confidences_batch: a list of tensors with length batch_size, where
590 each tensor has shape [num_gt_boxes_i, classification_target_size] and
591 num_gt_boxes_i is the number of boxes in the ith boxlist of
592 gt_box_batch. Note that in this tensor, 1 means explicit positive class,
593 -1 means explicit negative class, and 0 means implicit negative class.
594 gt_weights_batch: A list of 1-D tf.float32 tensors of shape
595 [num_gt_boxes_i] containing weights for groundtruth boxes.
596 unmatched_class_label: a float32 tensor with shape [d_1, d_2, ..., d_k]
597 which is consistent with the classification target for each
598 anchor (and can be empty for scalar targets). This shape must thus be
599 compatible with the groundtruth labels that are passed to the "assign"
600 function (which have shape [num_gt_boxes, d_1, d_2, ..., d_k]).
601 include_background_class: whether or not gt_class_confidences_batch includes
602 the background class.
603 implicit_class_weight: the weight assigned to implicit examples.
604
605 Returns:
606 batch_cls_targets: a tensor with shape [batch_size, num_anchors,
607 num_classes],
608 batch_cls_weights: a tensor with shape [batch_size, num_anchors,
609 num_classes],
610 batch_reg_targets: a tensor with shape [batch_size, num_anchors,
611 box_code_dimension]
612 batch_reg_weights: a tensor with shape [batch_size, num_anchors],
613 match: an int32 tensor of shape [batch_size, num_anchors] containing result
614 of anchor groundtruth matching. Each position in the tensor indicates an
615 anchor and holds the following meaning:
616 (1) if match[x, i] >= 0, anchor i is matched with groundtruth match[x, i].
617 (2) if match[x, i]=-1, anchor i is marked to be background .
618 (3) if match[x, i]=-2, anchor i is ignored since it is not background and
619 does not have sufficient overlap to call it a foreground.
620
621 Raises:
622 ValueError: if input list lengths are inconsistent, i.e.,
623 batch_size == len(gt_box_batch) == len(gt_class_targets_batch)
624 and batch_size == len(anchors_batch) unless anchors_batch is a single
625 BoxList, or if any element in gt_class_confidences_batch has rank > 2.
626 """
627 if not isinstance(anchors_batch, list):
628 anchors_batch = len(gt_box_batch) * [anchors_batch]
629 if not all(
630 isinstance(anchors, box_list.BoxList) for anchors in anchors_batch):
631 raise ValueError('anchors_batch must be a BoxList or list of BoxLists.')
632 if not (len(anchors_batch)
633 == len(gt_box_batch)
634 == len(gt_class_confidences_batch)):
635 raise ValueError('batch size incompatible with lengths of anchors_batch, '
636 'gt_box_batch and gt_class_confidences_batch.')
637
638 cls_targets_list = []
639 cls_weights_list = []
640 reg_targets_list = []
641 reg_weights_list = []
642 match_list = []
643 if gt_weights_batch is None:
644 gt_weights_batch = [None] * len(gt_class_confidences_batch)
645 for anchors, gt_boxes, gt_class_confidences, gt_weights in zip(
646 anchors_batch, gt_box_batch, gt_class_confidences_batch,
647 gt_weights_batch):
648
649 if (gt_class_confidences is not None and
650 len(gt_class_confidences.get_shape().as_list()) > 2):
651 raise ValueError('The shape of the class target is not supported. ',
652 gt_class_confidences.get_shape())
653
654 cls_targets, _, reg_targets, _, match = target_assigner.assign(
655 anchors, gt_boxes, gt_class_confidences, unmatched_class_label,
656 groundtruth_weights=gt_weights)
657
658 if include_background_class:
659 cls_targets_without_background = tf.slice(
660 cls_targets, [0, 1], [-1, -1])
661 else:
662 cls_targets_without_background = cls_targets
663
664 positive_mask = tf.greater(cls_targets_without_background, 0.0)
665 negative_mask = tf.less(cls_targets_without_background, 0.0)
666 explicit_example_mask = tf.logical_or(positive_mask, negative_mask)
667 positive_anchors = tf.reduce_any(positive_mask, axis=-1)
668
669 regression_weights = tf.cast(positive_anchors, dtype=tf.float32)
670 regression_targets = (
671 reg_targets * tf.expand_dims(regression_weights, axis=-1))
672 regression_weights_expanded = tf.expand_dims(regression_weights, axis=-1)
673
674 cls_targets_without_background = (
675 cls_targets_without_background *
676 (1 - tf.cast(negative_mask, dtype=tf.float32)))
677 cls_weights_without_background = ((1 - implicit_class_weight) * tf.cast(
678 explicit_example_mask, dtype=tf.float32) + implicit_class_weight)
679
680 if include_background_class:
681 cls_weights_background = (
682 (1 - implicit_class_weight) * regression_weights_expanded
683 + implicit_class_weight)
684 classification_weights = tf.concat(
685 [cls_weights_background, cls_weights_without_background], axis=-1)
686 cls_targets_background = 1 - regression_weights_expanded
687 classification_targets = tf.concat(
688 [cls_targets_background, cls_targets_without_background], axis=-1)
689 else:
690 classification_targets = cls_targets_without_background
691 classification_weights = cls_weights_without_background
692
693 cls_targets_list.append(classification_targets)
694 cls_weights_list.append(classification_weights)
695 reg_targets_list.append(regression_targets)
696 reg_weights_list.append(regression_weights)
697 match_list.append(match)
698 batch_cls_targets = tf.stack(cls_targets_list)
699 batch_cls_weights = tf.stack(cls_weights_list)
700 batch_reg_targets = tf.stack(reg_targets_list)
701 batch_reg_weights = tf.stack(reg_weights_list)
702 batch_match = tf.stack(match_list)
703 return (batch_cls_targets, batch_cls_weights, batch_reg_targets,
704 batch_reg_weights, batch_match)
