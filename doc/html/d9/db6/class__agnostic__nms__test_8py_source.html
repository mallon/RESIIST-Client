26 def test_class_agnostic_nms_select_with_shared_boxes(self):
27 boxes = tf.constant(
28 [[[0, 0, 1, 1]], [[0, 0.1, 1, 1.1]], [[0, -0.1, 1, 0.9]],
29 [[0, 10, 1, 11]], [[0, 10.1, 1, 11.1]], [[0, 100, 1, 101]],
30 [[0, 1000, 1, 1002]], [[0, 1000, 1, 1002.1]]], tf.float32)
31 scores = tf.constant([[.9, 0.01], [.75, 0.05], [.6, 0.01], [.95, 0],
32 [.5, 0.01], [.3, 0.01], [.01, .85], [.01, .5]])
33 score_thresh = 0.1
34 iou_thresh = .5
35 max_classes_per_detection = 1
36 max_output_size = 4
37
38 exp_nms_corners = [[0, 10, 1, 11], [0, 0, 1, 1], [0, 1000, 1, 1002],
39 [0, 100, 1, 101]]
40 exp_nms_scores = [.95, .9, .85, .3]
41 exp_nms_classes = [0, 0, 1, 0]
42
43 nms, _ = post_processing.class_agnostic_non_max_suppression(
44 boxes, scores, score_thresh, iou_thresh, max_classes_per_detection,
45 max_output_size)
46
47 with self.test_session() as sess:
48 nms_corners_output, nms_scores_output, nms_classes_output = sess.run([
