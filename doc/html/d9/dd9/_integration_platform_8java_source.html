418 private List<IndicatorDto> getTargetedDtos(String indicatorId)
419 {
420 List<IndicatorDto> indicatorDtos = new ArrayList<>();
421 List<IndicatorDto> dupplicateIndicatorDtos =
422 allSimulatedIndicators.stream()
423 .filter(confParam -> ((IndicatorDto) confParam.get(ResiistConstants.INDICATOR_DTO)).getElementId()
424 .equals(indicatorId))
425 .map(confParam -> ((IndicatorDto) confParam.get(ResiistConstants.INDICATOR_DTO)))
426 .collect(Collectors.toList());
427
428 for (IndicatorDto indicatorDto : dupplicateIndicatorDtos) {
429 if (!isAlreadyPresent(indicatorDto.getElementId(), indicatorDtos)) {
430 indicatorDtos.add(indicatorDto);
431 }
432 }
433
434 if (indicatorDtos.isEmpty()) {
435 List<String> correspondingCriterias = getCorrespondingCriterias(indicatorId);
436 indicatorDtos = correspondingCriterias.stream()
437 .map(this::getTargetedDtos)
438 .flatMap(List::stream)
439 .collect(Collectors.toList());
440
441 }
442
443 return indicatorDtos;

478 TreeIterator<EObject> iterator = modelResource.getAllContents();
479 while (iterator.hasNext()) {
480 EObject nextEObject = iterator.next();
481

483 EModelElement modelElement = (EModelElement) nextEObject;

485 String elementId = GMMUtil.getUUID(modelElement);

487 elementAndWidgetId.put(ResiistConstants.ELEMENT_IDENTIFIER, elementId);
488 ModelUtils.getName(modelElement)

490 name));
491 if (modelElement instanceof ModelImpl) {
492 elementAndWidgetId.put(ResiistConstants.SIMULATION_FILE_PATH, "");
493 } else if (ModelUtils.hasSimulationPath(modelElement)
494 || ModelUtils.hasGAMASimulationPath(modelElement))

497 ModelUtils.getSimulationFilePath(modelElement));
498 elementAndWidgetId.put(ResiistConstants.GAMA_SIMULATION_FILE_PATH,
499 ModelUtils.getGAMASimulationFilePath(modelElement));
500 }
501
502 elementAndWidgetIds.add(elementAndWidgetId);
503 }
504 }
505 }
