15"""Evaluates Visual Relations Detection(VRD) result evaluation on an image.
16
17Annotate each VRD result as true positives or false positive according to
18a predefined IOU ratio. Multi-class detection is supported by default.
19Based on the settings, per image evaluation is performed either on phrase
20detection subtask or on relation detection subtask.
21"""

33class PerImageVRDEvaluation(object):
34 """Evaluate vrd result of a single image."""

36 def __init__(self, matching_iou_threshold=0.5):
37 """Initialized PerImageVRDEvaluation by evaluation parameters.
38
39 Args:
40 matching_iou_threshold: A ratio of area intersection to union, which is
41 the threshold to consider whether a detection is true positive or not;
42 in phrase detection subtask.
43 """
44 self.matching_iou_threshold = matching_iou_threshold

46 def compute_detection_tp_fp(self, detected_box_tuples, detected_scores,
47 detected_class_tuples, groundtruth_box_tuples,
48 groundtruth_class_tuples):
49 """Evaluates VRD as being tp, fp from a single image.
50
51 Args:
52 detected_box_tuples: A numpy array of structures with shape [N,],
53 representing N tuples, each tuple containing the same number of named
54 bounding boxes.
55 Each box is of the format [y_min, x_min, y_max, x_max].
56 detected_scores: A float numpy array of shape [N,], representing
57 the confidence scores of the detected N object instances.
58 detected_class_tuples: A numpy array of structures shape [N,],
59 representing the class labels of the corresponding bounding boxes and
60 possibly additional classes.
61 groundtruth_box_tuples: A float numpy array of structures with the shape
62 [M,], representing M tuples, each tuple containing the same number
63 of named bounding boxes.
64 Each box is of the format [y_min, x_min, y_max, x_max].
65 groundtruth_class_tuples: A numpy array of structures shape [M,],
66 representing the class labels of the corresponding bounding boxes and
67 possibly additional classes.
68
69 Returns:
70 scores: A single numpy array with shape [N,], representing N scores
71 detected with object class, sorted in descentent order.
72 tp_fp_labels: A single boolean numpy array of shape [N,], representing N
73 True/False positive label, one label per tuple. The labels are sorted
74 so that the order of the labels matches the order of the scores.
75 result_mapping: A numpy array with shape [N,] with original index of each
76 entry.
77 """
78
79 scores, tp_fp_labels, result_mapping = self._compute_tp_fp(
80 detected_box_tuples=detected_box_tuples,
81 detected_scores=detected_scores,
82 detected_class_tuples=detected_class_tuples,
83 groundtruth_box_tuples=groundtruth_box_tuples,
84 groundtruth_class_tuples=groundtruth_class_tuples)
85
86 return scores, tp_fp_labels, result_mapping

88 def _compute_tp_fp(self, detected_box_tuples, detected_scores,
89 detected_class_tuples, groundtruth_box_tuples,
90 groundtruth_class_tuples):
91 """Labels as true/false positives detection tuples across all classes.
92
93 Args:
94 detected_box_tuples: A numpy array of structures with shape [N,],
95 representing N tuples, each tuple containing the same number of named
96 bounding boxes.
97 Each box is of the format [y_min, x_min, y_max, x_max]
98 detected_scores: A float numpy array of shape [N,], representing
99 the confidence scores of the detected N object instances.
100 detected_class_tuples: A numpy array of structures shape [N,],
101 representing the class labels of the corresponding bounding boxes and
102 possibly additional classes.
103 groundtruth_box_tuples: A float numpy array of structures with the shape
104 [M,], representing M tuples, each tuple containing the same number
105 of named bounding boxes.
106 Each box is of the format [y_min, x_min, y_max, x_max]
107 groundtruth_class_tuples: A numpy array of structures shape [M,],
108 representing the class labels of the corresponding bounding boxes and
109 possibly additional classes.
110
111 Returns:
112 scores: A single numpy array with shape [N,], representing N scores
113 detected with object class, sorted in descentent order.
114 tp_fp_labels: A single boolean numpy array of shape [N,], representing N
115 True/False positive label, one label per tuple. The labels are sorted
116 so that the order of the labels matches the order of the scores.
117 result_mapping: A numpy array with shape [N,] with original index of each
118 entry.
119 """
120 unique_gt_tuples = np.unique(
121 np.concatenate((groundtruth_class_tuples, detected_class_tuples)))
122 result_scores = []
123 result_tp_fp_labels = []
124 result_mapping = []
125
126 for unique_tuple in unique_gt_tuples:
127 detections_selector = (detected_class_tuples == unique_tuple)
128 gt_selector = (groundtruth_class_tuples == unique_tuple)
129
130 selector_mapping = np.where(detections_selector)[0]
131
132 detection_scores_per_tuple = detected_scores[detections_selector]
133 detection_box_per_tuple = detected_box_tuples[detections_selector]
134
135 sorted_indices = np.argsort(detection_scores_per_tuple)
136 sorted_indices = sorted_indices[::-1]
137
138 tp_fp_labels = self._compute_tp_fp_for_single_class(
139 detected_box_tuples=detection_box_per_tuple[sorted_indices],
140 groundtruth_box_tuples=groundtruth_box_tuples[gt_selector])
141 result_scores.append(detection_scores_per_tuple[sorted_indices])
142 result_tp_fp_labels.append(tp_fp_labels)
143 result_mapping.append(selector_mapping[sorted_indices])
144
145 if result_scores:
146 result_scores = np.concatenate(result_scores)
147 result_tp_fp_labels = np.concatenate(result_tp_fp_labels)
148 result_mapping = np.concatenate(result_mapping)
149 else:
150 result_scores = np.array([], dtype=float)
151 result_tp_fp_labels = np.array([], dtype=bool)
152 result_mapping = np.array([], dtype=int)
153
154 sorted_indices = np.argsort(result_scores)
155 sorted_indices = sorted_indices[::-1]
156
157 return result_scores[sorted_indices], result_tp_fp_labels[
158 sorted_indices], result_mapping[sorted_indices]

160 def _get_overlaps_and_scores_relation_tuples(self, detected_box_tuples,
161 groundtruth_box_tuples):
162 """Computes overlaps and scores between detected and groundtruth tuples.
163
164 Both detections and groundtruth boxes have the same class tuples.
165
166 Args:
167 detected_box_tuples: A numpy array of structures with shape [N,],
168 representing N tuples, each tuple containing the same number of named
169 bounding boxes.
170 Each box is of the format [y_min, x_min, y_max, x_max]
171 groundtruth_box_tuples: A float numpy array of structures with the shape
172 [M,], representing M tuples, each tuple containing the same number
173 of named bounding boxes.
174 Each box is of the format [y_min, x_min, y_max, x_max]
175
176 Returns:
177 result_iou: A float numpy array of size
178 [num_detected_tuples, num_gt_box_tuples].
179 """
180
181 result_iou = np.ones(
182 (detected_box_tuples.shape[0], groundtruth_box_tuples.shape[0]),
183 dtype=float)
184 for field in detected_box_tuples.dtype.fields:
185 detected_boxlist_field = np_box_list.BoxList(detected_box_tuples[field])
186 gt_boxlist_field = np_box_list.BoxList(groundtruth_box_tuples[field])
187 iou_field = np_box_list_ops.iou(detected_boxlist_field, gt_boxlist_field)
188 result_iou = np.minimum(iou_field, result_iou)
189 return result_iou

191 def _compute_tp_fp_for_single_class(self, detected_box_tuples,
192 groundtruth_box_tuples):
193 """Labels boxes detected with the same class from the same image as tp/fp.
194
195 Detection boxes are expected to be already sorted by score.
196 Args:
197 detected_box_tuples: A numpy array of structures with shape [N,],
198 representing N tuples, each tuple containing the same number of named
199 bounding boxes.
200 Each box is of the format [y_min, x_min, y_max, x_max]
201 groundtruth_box_tuples: A float numpy array of structures with the shape
202 [M,], representing M tuples, each tuple containing the same number
203 of named bounding boxes.
204 Each box is of the format [y_min, x_min, y_max, x_max]
205
206 Returns:
207 tp_fp_labels: a boolean numpy array indicating whether a detection is a
208 true positive.
209 """
210 if detected_box_tuples.size == 0:
211 return np.array([], dtype=bool)
212
213 min_iou = self._get_overlaps_and_scores_relation_tuples(
214 detected_box_tuples, groundtruth_box_tuples)
215
216 num_detected_tuples = detected_box_tuples.shape[0]
217 tp_fp_labels = np.zeros(num_detected_tuples, dtype=bool)
218
219 if min_iou.shape[1] > 0:
220 max_overlap_gt_ids = np.argmax(min_iou, axis=1)
221 is_gt_tuple_detected = np.zeros(min_iou.shape[1], dtype=bool)
222 for i in range(num_detected_tuples):
223 gt_id = max_overlap_gt_ids[i]
224 if min_iou[i, gt_id] >= self.matching_iou_threshold:
225 if not is_gt_tuple_detected[gt_id]:
226 tp_fp_labels[i] = True
227 is_gt_tuple_detected[gt_id] = True
228
229 return tp_fp_labels
