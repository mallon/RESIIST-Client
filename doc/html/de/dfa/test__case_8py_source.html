36 def execute_tpu(self, graph_fn, inputs):
37 """Constructs the graph, executes it on TPU and returns the result.
38
39 Args:
40 graph_fn: a callable that constructs the tensorflow graph to test. The
41 arguments of this function should correspond to `inputs`.
42 inputs: a list of numpy arrays to feed input to the computation graph.
43
44 Returns:
45 A list of numpy arrays or a scalar returned from executing the tensorflow
46 graph.
47 """
48 with self.test_session(graph=tf.Graph()) as sess:
49 placeholders = [tf.placeholder_with_default(v, v.shape) for v in inputs]
50 tpu_computation = tpu.rewrite(graph_fn, placeholders)
51 sess.run(tpu.initialize_system())
52 sess.run([tf.global_variables_initializer(), tf.tables_initializer(),
53 tf.local_variables_initializer()])
54 materialized_results = sess.run(tpu_computation,
55 feed_dict=dict(zip(placeholders, inputs)))
56 sess.run(tpu.shutdown_system())
57 if (hasattr(materialized_results, '__len__') and
58 len(materialized_results) == 1 and
59 (isinstance(materialized_results, list) or
60 isinstance(materialized_results, tuple))):
61 materialized_results = materialized_results[0]
62 return materialized_results

64 def execute_cpu(self, graph_fn, inputs):
65 """Constructs the graph, executes it on CPU and returns the result.
66
67 Args:
68 graph_fn: a callable that constructs the tensorflow graph to test. The
69 arguments of this function should correspond to `inputs`.
70 inputs: a list of numpy arrays to feed input to the computation graph.
71
72 Returns:
73 A list of numpy arrays or a scalar returned from executing the tensorflow
74 graph.
75 """
76 with self.test_session(graph=tf.Graph()) as sess:
77 placeholders = [tf.placeholder_with_default(v, v.shape) for v in inputs]
78 results = graph_fn(*placeholders)
79 sess.run([tf.global_variables_initializer(), tf.tables_initializer(),
80 tf.local_variables_initializer()])
81 materialized_results = sess.run(results, feed_dict=dict(zip(placeholders,
82 inputs)))
83
84 if (hasattr(materialized_results, '__len__') and
85 len(materialized_results) == 1 and
86 (isinstance(materialized_results, list) or
87 isinstance(materialized_results, tuple))):
88 materialized_results = materialized_results[0]
89 return materialized_results

91 def execute(self, graph_fn, inputs):
92 """Constructs the graph, creates a test session and returns the results.
93
94 The graph is executed either on TPU or CPU based on the `tpu_test` flag.
95
96 Args:
97 graph_fn: a callable that constructs the tensorflow graph to test. The
98 arguments of this function should correspond to `inputs`.
99 inputs: a list of numpy arrays to feed input to the computation graph.
100
101 Returns:
102 A list of numpy arrays or a scalar returned from executing the tensorflow
103 graph.
104 """
105 if FLAGS.tpu_test:
106 return self.execute_tpu(graph_fn, inputs)
107 else:
108 return self.execute_cpu(graph_fn, inputs)
