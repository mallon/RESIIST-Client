16"""Bounding Box List definition.
17
18BoxList represents a list of bounding boxes as tensorflow
19tensors, where each bounding box is represented as a row of 4 numbers,
20[y_min, x_min, y_max, x_max]. It is assumed that all bounding boxes
21within a given list correspond to a single image. See also
22box_list_ops.py for common box related operations (such as area, iou, etc).
23
24Optionally, users can add additional related fields (such as weights).
25We assume the following things to be true about fields:
26* they correspond to boxes in the box_list along the 0th dimension
27* they have inferrable rank at graph construction time
28* all dimensions except for possibly the 0th can be inferred
29 (i.e., not None) at graph construction time.
30
31Some other notes:
32 * Following tensorflow conventions, we use height, width ordering,
33 and correspondingly, y,x (or ymin, xmin, ymax, xmax) ordering
34 * Tensors are always provided as (flat) [N, 4] tensors.
35"""
36
37import tensorflow as tf
38
39from object_detection.utils import shape_utils
40
41
42class BoxList(object):
43 """Box collection."""
44
45 def __init__(self, boxes):
46 """Constructs box collection.
47
48 Args:
49 boxes: a tensor of shape [N, 4] representing box corners
50
51 Raises:
52 ValueError: if invalid dimensions for bbox data or if bbox data is not in
53 float32 format.
54 """
55 if len(boxes.get_shape()) != 2 or boxes.get_shape()[-1] != 4:
56 raise ValueError('Invalid dimensions for box data.')
57 if boxes.dtype != tf.float32:
58 raise ValueError('Invalid tensor type: should be tf.float32')
59 self.data = {'boxes': boxes}
60
61 def num_boxes(self):
62 """Returns number of boxes held in collection.
63
64 Returns:
65 a tensor representing the number of boxes held in the collection.
66 """
67 return tf.shape(self.data['boxes'])[0]
68
69 def num_boxes_static(self):
70 """Returns number of boxes held in collection.
71
72 This number is inferred at graph construction time rather than run-time.
73
74 Returns:
75 Number of boxes held in collection (integer) or None if this is not
76 inferrable at graph construction time.
77 """
78 return shape_utils.get_dim_as_int(self.data['boxes'].get_shape()[0])
79
80 def get_all_fields(self):
81 """Returns all fields."""
82 return self.data.keys()
83
84 def get_extra_fields(self):
85 """Returns all non-box fields (i.e., everything not named 'boxes')."""
86 return [k for k in self.data.keys() if k != 'boxes']
87
88 def add_field(self, field, field_data):
89 """Add field to box list.
90
91 This method can be used to add related box data such as
92 weights/labels, etc.
93
94 Args:
95 field: a string key to access the data via `get`
96 field_data: a tensor containing the data to store in the BoxList
97 """
98 self.data[field] = field_data
99
100 def has_field(self, field):
101 return field in self.data
102
103 def get(self):
104 """Convenience function for accessing box coordinates.
105
106 Returns:
107 a tensor with shape [N, 4] representing box coordinates.
108 """
109 return self.get_field('boxes')
110
111 def set(self, boxes):
112 """Convenience function for setting box coordinates.
113
114 Args:
115 boxes: a tensor of shape [N, 4] representing box corners
116
117 Raises:
118 ValueError: if invalid dimensions for bbox data
119 """
120 if len(boxes.get_shape()) != 2 or boxes.get_shape()[-1] != 4:
121 raise ValueError('Invalid dimensions for box data.')
122 self.data['boxes'] = boxes
123
124 def get_field(self, field):
125 """Accesses a box collection and associated fields.
126
127 This function returns specified field with object; if no field is specified,
128 it returns the box coordinates.
129
130 Args:
131 field: this optional string parameter can be used to specify
132 a related field to be accessed.
133
134 Returns:
135 a tensor representing the box collection or an associated field.
136
137 Raises:
138 ValueError: if invalid field
139 """
140 if not self.has_field(field):
141 raise ValueError('field ' + str(field) + ' does not exist')
142 return self.data[field]
143
144 def set_field(self, field, value):
145 """Sets the value of a field.
146
147 Updates the field of a box_list with a given value.
148
149 Args:
150 field: (string) name of the field to set value.
151 value: the value to assign to the field.
152
153 Raises:
154 ValueError: if the box_list does not have specified field.
155 """
156 if not self.has_field(field):
157 raise ValueError('field %s does not exist' % field)
158 self.data[field] = value
159
160 def get_center_coordinates_and_sizes(self, scope=None):
161 """Computes the center coordinates, height and width of the boxes.
162
163 Args:
164 scope: name scope of the function.
165
166 Returns:
167 a list of 4 1-D tensors [ycenter, xcenter, height, width].
168 """
169 with tf.name_scope(scope, 'get_center_coordinates_and_sizes'):
170 box_corners = self.get()
171 ymin, xmin, ymax, xmax = tf.unstack(tf.transpose(box_corners))
172 width = xmax - xmin
173 height = ymax - ymin
174 ycenter = ymin + height / 2.
175 xcenter = xmin + width / 2.
176 return [ycenter, xcenter, height, width]
177
178 def transpose_coordinates(self, scope=None):
179 """Transpose the coordinate representation in a boxlist.
180
181 Args:
182 scope: name scope of the function.
183 """
184 with tf.name_scope(scope, 'transpose_coordinates'):
185 y_min, x_min, y_max, x_max = tf.split(
186 value=self.get(), num_or_size_splits=4, axis=1)
187 self.set(tf.concat([x_min, y_min, x_max, y_max], 1))
188
189 def as_tensor_dict(self, fields=None):
190 """Retrieves specified fields as a dictionary of tensors.
191
192 Args:
193 fields: (optional) list of fields to return in the dictionary.
194 If None (default), all fields are returned.
195
196 Returns:
197 tensor_dict: A dictionary of tensors specified by fields.
198
199 Raises:
200 ValueError: if specified field is not contained in boxlist.
201 """
202 tensor_dict = {}
203 if fields is None:
204 fields = self.get_all_fields()
205 for field in fields:
206 if not self.has_field(field):
207 raise ValueError('boxlist must contain all specified fields')
208 tensor_dict[field] = self.get_field(field)
209 return tensor_dict
