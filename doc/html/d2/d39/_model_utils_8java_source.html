160 public static Optional<String> getName(EObject eObject)
161 {
162 EStructuralFeature nameFeature = eObject.eClass().getEStructuralFeature("name");
163 if (nameFeature == null)
164 return Optional.empty();
165
166 Object nameValue = eObject.eGet(nameFeature);
167 if (nameValue == null)
168 return Optional.empty();
169
170 if (nameValue instanceof EList<?>) {
171 EList<?> nameValues = ((EList<?>) eObject.eGet(nameFeature));
172 if (nameValues.isEmpty()) {
173 return Optional.empty();
174 }
175 EObject firstNameLiteral = (EObject) nameValues.get(0);
176 EStructuralFeature valueFeature =
177 firstNameLiteral.eClass().getEStructuralFeature("value");
178 if (valueFeature == null)
179 return Optional.empty();
180
181 return Optional.ofNullable((String) firstNameLiteral.eGet(valueFeature));
182 }
183
184 return Optional.ofNullable((String) nameValue);
185 }
