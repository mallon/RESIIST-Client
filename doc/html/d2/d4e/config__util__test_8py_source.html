37def _write_config(config, config_path):
38 """Writes a config object to disk."""
39 config_text = text_format.MessageToString(config)
40 with tf.gfile.Open(config_path, "wb") as f:
41 f.write(config_text)

44def _update_optimizer_with_constant_learning_rate(optimizer, learning_rate):
45 """Adds a new constant learning rate."""
46 constant_lr = optimizer.learning_rate.constant_learning_rate
47 constant_lr.learning_rate = learning_rate

50def _update_optimizer_with_exponential_decay_learning_rate(
51 optimizer, learning_rate):
52 """Adds a new exponential decay learning rate."""
53 exponential_lr = optimizer.learning_rate.exponential_decay_learning_rate
54 exponential_lr.initial_learning_rate = learning_rate

57def _update_optimizer_with_manual_step_learning_rate(
58 optimizer, initial_learning_rate, learning_rate_scaling):
59 """Adds a learning rate schedule."""
60 manual_lr = optimizer.learning_rate.manual_step_learning_rate
61 manual_lr.initial_learning_rate = initial_learning_rate
62 for i in range(3):
63 schedule = manual_lr.schedule.add()
64 schedule.learning_rate = initial_learning_rate * learning_rate_scaling**i

67def _update_optimizer_with_cosine_decay_learning_rate(
68 optimizer, learning_rate, warmup_learning_rate):
69 """Adds a new cosine decay learning rate."""
70 cosine_lr = optimizer.learning_rate.cosine_decay_learning_rate
71 cosine_lr.learning_rate_base = learning_rate
72 cosine_lr.warmup_learning_rate = warmup_learning_rate

207 def _assertOptimizerWithNewLearningRate(self, optimizer_name):
208 """Asserts successful updating of all learning rate schemes."""
209 original_learning_rate = 0.7
210 learning_rate_scaling = 0.1
211 warmup_learning_rate = 0.07
212 hparams = tf.contrib.training.HParams(learning_rate=0.15)
213 pipeline_config_path = os.path.join(self.get_temp_dir(), "pipeline.config")
214
215 # Constant learning rate.
216 pipeline_config = pipeline_pb2.TrainEvalPipelineConfig()
217 optimizer = getattr(pipeline_config.train_config.optimizer, optimizer_name)
218 _update_optimizer_with_constant_learning_rate(optimizer,
219 original_learning_rate)
220 _write_config(pipeline_config, pipeline_config_path)
221
222 configs = config_util.get_configs_from_pipeline_file(pipeline_config_path)
223 configs = config_util.merge_external_params_with_configs(configs, hparams)
224 optimizer = getattr(configs["train_config"].optimizer, optimizer_name)
225 constant_lr = optimizer.learning_rate.constant_learning_rate
226 self.assertAlmostEqual(hparams.learning_rate, constant_lr.learning_rate)
227
228 # Exponential decay learning rate.
229 pipeline_config = pipeline_pb2.TrainEvalPipelineConfig()
230 optimizer = getattr(pipeline_config.train_config.optimizer, optimizer_name)
231 _update_optimizer_with_exponential_decay_learning_rate(
232 optimizer, original_learning_rate)
233 _write_config(pipeline_config, pipeline_config_path)
234
235 configs = config_util.get_configs_from_pipeline_file(pipeline_config_path)
236 configs = config_util.merge_external_params_with_configs(configs, hparams)
237 optimizer = getattr(configs["train_config"].optimizer, optimizer_name)
238 exponential_lr = optimizer.learning_rate.exponential_decay_learning_rate
239 self.assertAlmostEqual(hparams.learning_rate,
240 exponential_lr.initial_learning_rate)
241
242 # Manual step learning rate.
243 pipeline_config = pipeline_pb2.TrainEvalPipelineConfig()
244 optimizer = getattr(pipeline_config.train_config.optimizer, optimizer_name)
245 _update_optimizer_with_manual_step_learning_rate(
246 optimizer, original_learning_rate, learning_rate_scaling)
247 _write_config(pipeline_config, pipeline_config_path)
248
249 configs = config_util.get_configs_from_pipeline_file(pipeline_config_path)
250 configs = config_util.merge_external_params_with_configs(configs, hparams)
251 optimizer = getattr(configs["train_config"].optimizer, optimizer_name)
252 manual_lr = optimizer.learning_rate.manual_step_learning_rate
253 self.assertAlmostEqual(hparams.learning_rate,
254 manual_lr.initial_learning_rate)
255 for i, schedule in enumerate(manual_lr.schedule):
256 self.assertAlmostEqual(hparams.learning_rate * learning_rate_scaling**i,
257 schedule.learning_rate)
258
259 # Cosine decay learning rate.
260 pipeline_config = pipeline_pb2.TrainEvalPipelineConfig()
261 optimizer = getattr(pipeline_config.train_config.optimizer, optimizer_name)
262 _update_optimizer_with_cosine_decay_learning_rate(optimizer,
263 original_learning_rate,
264 warmup_learning_rate)
265 _write_config(pipeline_config, pipeline_config_path)
266
267 configs = config_util.get_configs_from_pipeline_file(pipeline_config_path)
268 configs = config_util.merge_external_params_with_configs(configs, hparams)
269 optimizer = getattr(configs["train_config"].optimizer, optimizer_name)
270 cosine_lr = optimizer.learning_rate.cosine_decay_learning_rate
271
272 self.assertAlmostEqual(hparams.learning_rate, cosine_lr.learning_rate_base)
273 warmup_scale_factor = warmup_learning_rate / original_learning_rate
274 self.assertAlmostEqual(hparams.learning_rate * warmup_scale_factor,
275 cosine_lr.warmup_learning_rate)
