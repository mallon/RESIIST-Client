288 private void displayReportedInfluencedIndicatorValues()
289 {
290 IntegrationPlatform ip = IntegrationPlatform.getInstance();
291 Map<String, Map<String, String>> alternAnfPerf = decisionMaking.getAlternativeAndValues();
292 for (Value perfReport : decisionMaking.getReportPerformances()) {
293 String perfReportName = ModelUtils.getArchimateName(perfReport);
294 EMap<String, String> perfReportDetails = ModelUtils.getElementDetails(perfReport);
295 String alterName = perfReportDetails.get(ResiistConstants.ALTERNATIVE);
296 String criterionDec = perfReportDetails.get(ResiistConstants.DECISION_CRITERION);
297 Map<String, String> criterionAndPerf = alternAnfPerf.get(alterName);
298 String perfReportValue = criterionAndPerf.get(criterionDec);
299 String perfReportOldValue = ModelUtils.getArchimateName(perfReport);
300 ip.setValueForPerf(perfReport, perfReportOldValue, Double.valueOf(perfReportValue));
