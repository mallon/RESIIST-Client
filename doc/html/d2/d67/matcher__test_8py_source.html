51 (num_matched_columns_out, num_unmatched_columns_out,
52 num_ignored_columns_out, num_matched_rows_out) = sess.run(

133 def test_all_columns_accounted_for(self):
134 # Note: deliberately setting to small number so not always
135 # all possibilities appear (matched, unmatched, ignored)
136 num_matches = 10
137 match_results = tf.random_uniform(
138 [num_matches], minval=-2, maxval=5, dtype=tf.int32)
139 match = matcher.Match(match_results)
140 matched_column_indices = match.matched_column_indices()
141 unmatched_column_indices = match.unmatched_column_indices()
142 ignored_column_indices = match.ignored_column_indices()
143 with self.test_session() as sess:
144 matched, unmatched, ignored = sess.run([
145 matched_column_indices, unmatched_column_indices,
146 ignored_column_indices
147 ])
148 all_indices = np.hstack((matched, unmatched, ignored))
149 all_indices_sorted = np.sort(all_indices)
150 self.assertAllEqual(all_indices_sorted,
151 np.arange(num_matches, dtype=np.int32))

153 def test_scalar_gather_based_on_match(self):

180 def test_multidimensional_gather_based_on_match_with_matmul_gather_op(self):
181 match_results = tf.constant([1, -1, -2])
182 input_tensor = tf.constant([[0, 0.5, 0, 0.5], [0, 0, 0.5, 0.5]],
183 dtype=tf.float32)
184 expected_gathered_tensor = [[0, 0, 0.5, 0.5], [0, 0, 0, 0], [0, 0, 0, 0]]
185 match = matcher.Match(match_results, use_matmul_gather=True)

189 self.assertEqual(gathered_tensor.dtype, tf.float32)
190 with self.test_session() as sess:
191 self.assertTrue(
192 all([op.name is not 'Gather' for op in sess.graph.get_operations()]))
