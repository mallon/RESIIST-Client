31 def test_prefetch_tensors_with_fully_defined_shapes(self):
32 with self.test_session() as sess:
33 batch_size = 10
34 image_size = 32
35 num_batches = 5
36 examples = tf.Variable(tf.constant(0, dtype=tf.int64))
37 counter = examples.count_up_to(num_batches)
38 image = tf.random_normal([batch_size, image_size,
39 image_size, 3],
40 dtype=tf.float32,
41 name='images')
42 label = tf.random_uniform([batch_size, 1], 0, 10,
43 dtype=tf.int32, name='labels')
44
45 prefetch_queue = prefetcher.prefetch(tensor_dict={'counter': counter,
46 'image': image,
47 'label': label},
48 capacity=100)
49 tensor_dict = prefetch_queue.dequeue()
50
51 self.assertAllEqual(tensor_dict['image'].get_shape().as_list(),
52 [batch_size, image_size, image_size, 3])
53 self.assertAllEqual(tensor_dict['label'].get_shape().as_list(),
54 [batch_size, 1])
55
56 tf.initialize_all_variables().run()
57 with slim.queues.QueueRunners(sess):
58 for _ in range(num_batches):
59 results = sess.run(tensor_dict)
60 self.assertEquals(results['image'].shape,
61 (batch_size, image_size, image_size, 3))
62 self.assertEquals(results['label'].shape, (batch_size, 1))
63 with self.assertRaises(tf.errors.OutOfRangeError):
64 sess.run(tensor_dict)
