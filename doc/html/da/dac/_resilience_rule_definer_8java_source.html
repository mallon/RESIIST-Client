85 private Float evaluateResilience(Float fn, Float fnmax, Float fmax, Float fnmin, Float fmin)
86 {
87 if (fn >= fmax || fn <= fmin)
88 return Float.valueOf(0);
89
90 if (fn > fnmax)
91 return 1 - ((fn - fnmax) / (fmax - fnmax));
92
93 if (fn < fnmin)
94 return 1 - ((fnmin - fn) / (fnmin - fmin));
95
96 return Float.valueOf(1);
