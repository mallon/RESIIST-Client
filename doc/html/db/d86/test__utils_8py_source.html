148def create_diagonal_gradient_image(height, width, depth):
149 """Creates pyramid image. Useful for testing.
150
151 For example, pyramid_image(5, 6, 1) looks like:
152 # [[[ 5. 4. 3. 2. 1. 0.]
153 # [ 6. 5. 4. 3. 2. 1.]
154 # [ 7. 6. 5. 4. 3. 2.]
155 # [ 8. 7. 6. 5. 4. 3.]
156 # [ 9. 8. 7. 6. 5. 4.]]]
157
158 Args:
159 height: height of image
160 width: width of image
161 depth: depth of image
162
163 Returns:
164 pyramid image
165 """
166 row = np.arange(height)
167 col = np.arange(width)[::-1]
168 image_layer = np.expand_dims(row, 1) + col
169 image_layer = np.expand_dims(image_layer, 2)
170
171 image = image_layer
172 for i in range(1, depth):
173 image = np.concatenate((image, image_layer * pow(10, i)), 2)
174
175 return image.astype(np.float32)

178def create_random_boxes(num_boxes, max_height, max_width):
179 """Creates random bounding boxes of specific maximum height and width.
180
181 Args:
182 num_boxes: number of boxes.
183 max_height: maximum height of boxes.
184 max_width: maximum width of boxes.
185
186 Returns:
187 boxes: numpy array of shape [num_boxes, 4]. Each row is in form
188 [y_min, x_min, y_max, x_max].
189 """
190
191 y_1 = np.random.uniform(size=(1, num_boxes)) * max_height
192 y_2 = np.random.uniform(size=(1, num_boxes)) * max_height
193 x_1 = np.random.uniform(size=(1, num_boxes)) * max_width
194 x_2 = np.random.uniform(size=(1, num_boxes)) * max_width
195
196 boxes = np.zeros(shape=(num_boxes, 4))
197 boxes[:, 0] = np.minimum(y_1, y_2)
198 boxes[:, 1] = np.minimum(x_1, x_2)
199 boxes[:, 2] = np.maximum(y_1, y_2)
200 boxes[:, 3] = np.maximum(x_1, x_2)
201
202 return boxes.astype(np.float32)

205def first_rows_close_as_set(a, b, k=None, rtol=1e-6, atol=1e-6):
206 """Checks if first K entries of two lists are close, up to permutation.
207
208 Inputs to this assert are lists of items which can be compared via
209 numpy.allclose(...) and can be sorted.
210
211 Args:
212 a: list of items which can be compared via numpy.allclose(...) and are
213 sortable.
214 b: list of items which can be compared via numpy.allclose(...) and are
215 sortable.
216 k: a non-negative integer. If not provided, k is set to be len(a).
217 rtol: relative tolerance.
218 atol: absolute tolerance.
219
220 Returns:
221 boolean, True if input lists a and b have the same length and
222 the first k entries of the inputs satisfy numpy.allclose() after
223 sorting entries.
224 """
225 if not isinstance(a, list) or not isinstance(b, list) or len(a) != len(b):
226 return False
227 if not k:
228 k = len(a)
229 k = min(k, len(a))
230 a_sorted = sorted(a[:k])
231 b_sorted = sorted(b[:k])
232 return all([
233 np.allclose(entry_a, entry_b, rtol, atol)
234 for (entry_a, entry_b) in zip(a_sorted, b_sorted)
235 ])
