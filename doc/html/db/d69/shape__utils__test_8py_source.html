149 def test_with_dynamic_shape(self):
150 def fn(input_tensor):
151 return tf.reduce_sum(input_tensor)
152 input_tensor = tf.placeholder(tf.float32, shape=(None, 2))
153 map_fn_output = shape_utils.static_or_dynamic_map_fn(fn, input_tensor)
154
155 op_names = [op.name for op in tf.get_default_graph().get_operations()]
156 self.assertTrue(any(['map' == op_name[:3] for op_name in op_names]))
157
158 with self.test_session() as sess:
159 result1 = sess.run(
160 map_fn_output, feed_dict={
161 input_tensor: [[1, 2], [3, 1], [0, 4]]})
162 result2 = sess.run(
163 map_fn_output, feed_dict={
164 input_tensor: [[-1, 1], [0, 9]]})
165 self.assertAllEqual(result1, [3, 4, 4])
166 self.assertAllEqual(result2, [0, 9])
