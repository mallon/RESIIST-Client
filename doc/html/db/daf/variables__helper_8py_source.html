35def filter_variables(variables, filter_regex_list, invert=False):
36 """Filters out the variables matching the filter_regex.
37
38 Filter out the variables whose name matches the any of the regular
39 expressions in filter_regex_list and returns the remaining variables.
40 Optionally, if invert=True, the complement set is returned.
41
42 Args:
43 variables: a list of tensorflow variables.
44 filter_regex_list: a list of string regular expressions.
45 invert: (boolean). If True, returns the complement of the filter set; that
46 is, all variables matching filter_regex are kept and all others discarded.
47
48 Returns:
49 a list of filtered variables.
50 """
51 kept_vars = []
52 variables_to_ignore_patterns = list([fre for fre in filter_regex_list if fre])
53 for var in variables:
54 add = True
55 for pattern in variables_to_ignore_patterns:
56 if re.match(pattern, var.op.name):
57 add = False
58 break
59 if add != invert:
60 kept_vars.append(var)
61 return kept_vars

64def multiply_gradients_matching_regex(grads_and_vars, regex_list, multiplier):
65 """Multiply gradients whose variable names match a regular expression.
66
67 Args:
68 grads_and_vars: A list of gradient to variable pairs (tuples).
69 regex_list: A list of string regular expressions.
70 multiplier: A (float) multiplier to apply to each gradient matching the
71 regular expression.
72
73 Returns:
74 grads_and_vars: A list of gradient to variable pairs (tuples).
75 """
76 variables = [pair[1] for pair in grads_and_vars]
77 matching_vars = filter_variables(variables, regex_list, invert=True)
78 for var in matching_vars:
79 logging.info('Applying multiplier %f to variable [%s]',
80 multiplier, var.op.name)
81 grad_multipliers = {var: float(multiplier) for var in matching_vars}
82 return slim.learning.multiply_gradients(grads_and_vars,
83 grad_multipliers)

86def freeze_gradients_matching_regex(grads_and_vars, regex_list):
87 """Freeze gradients whose variable names match a regular expression.
88
89 Args:
90 grads_and_vars: A list of gradient to variable pairs (tuples).
91 regex_list: A list of string regular expressions.
92
93 Returns:
94 grads_and_vars: A list of gradient to variable pairs (tuples) that do not
95 contain the variables and gradients matching the regex.
96 """
97 variables = [pair[1] for pair in grads_and_vars]
98 matching_vars = filter_variables(variables, regex_list, invert=True)
99 kept_grads_and_vars = [pair for pair in grads_and_vars
100 if pair[1] not in matching_vars]
101 for var in matching_vars:
102 logging.info('Freezing variable [%s]', var.op.name)
103 return kept_grads_and_vars

106def get_variables_available_in_checkpoint(variables,
107 checkpoint_path,
108 include_global_step=True):
109 """Returns the subset of variables available in the checkpoint.
110
111 Inspects given checkpoint and returns the subset of variables that are
112 available in it.
113
114 TODO(rathodv): force input and output to be a dictionary.
115
116 Args:
117 variables: a list or dictionary of variables to find in checkpoint.
118 checkpoint_path: path to the checkpoint to restore variables from.
119 include_global_step: whether to include `global_step` variable, if it
120 exists. Default True.
121
122 Returns:
123 A list or dictionary of variables.
124 Raises:
125 ValueError: if `variables` is not a list or dict.
126 """
127 if isinstance(variables, list):
128 variable_names_map = {}
129 for variable in variables:
130 if isinstance(variable, tf_variables.PartitionedVariable):
131 name = variable.name
132 else:
133 name = variable.op.name
134 variable_names_map[name] = variable
135 elif isinstance(variables, dict):
136 variable_names_map = variables
137 else:
138 raise ValueError('`variables` is expected to be a list or dict.')
139 ckpt_reader = tf.train.NewCheckpointReader(checkpoint_path)
140 ckpt_vars_to_shape_map = ckpt_reader.get_variable_to_shape_map()
141 if not include_global_step:
142 ckpt_vars_to_shape_map.pop(tf.GraphKeys.GLOBAL_STEP, None)
143 vars_in_ckpt = {}
144 for variable_name, variable in sorted(variable_names_map.items()):
145 if variable_name in ckpt_vars_to_shape_map:
146 if ckpt_vars_to_shape_map[variable_name] == variable.shape.as_list():
147 vars_in_ckpt[variable_name] = variable
148 else:
149 logging.warning('Variable [%s] is available in checkpoint, but has an '
150 'incompatible shape with model variable. Checkpoint '
151 'shape: [%s], model variable shape: [%s]. This '
152 'variable will not be initialized from the checkpoint.',
153 variable_name, ckpt_vars_to_shape_map[variable_name],
154 variable.shape.as_list())
155 else:
156 logging.warning('Variable [%s] is not available in checkpoint',
157 variable_name)
158 if isinstance(variables, list):
159 return list(vars_in_ckpt.values())
160 return vars_in_ckpt

164 """If not executing eagerly, returns tf.global_variables().

169 If executing eagerly, use a Keras model's .variables property instead.
170
171 Returns:
172 The result of tf.global_variables()
173 """
174 with tf.init_scope():
175 if tf.executing_eagerly():
176 raise ValueError("Global variables collection is not tracked when "
