16"""Matcher interface and Match class.
17
18This module defines the Matcher interface and the Match object. The job of the
19matcher is to match row and column indices based on the similarity matrix and
20other optional parameters. Each column is matched to at most one row. There
21are three possibilities for the matching:
22
231) match: A column matches a row.
242) no_match: A column does not match any row.
253) ignore: A column that is neither 'match' nor no_match.
26
27The ignore case is regularly encountered in object detection: when an anchor has
28a relatively small overlap with a ground-truth box, one neither wants to
29consider this box a positive example (match) nor a negative example (no match).
30
31The Match class is used to store the match results and it provides simple apis
32to query the results.
33"""

45class Match(object):
46 """Class to store results from the matcher.
47
48 This class is used to store the results from the matcher. It provides
49 convenient methods to query the matching results.
50 """
51
52 def __init__(self, match_results, use_matmul_gather=False):
53 """Constructs a Match object.
54
55 Args:
56 match_results: Integer tensor of shape [N] with (1) match_results[i]>=0,
57 meaning that column i is matched with row match_results[i].
58 (2) match_results[i]=-1, meaning that column i is not matched.
59 (3) match_results[i]=-2, meaning that column i is ignored.
60 use_matmul_gather: Use matrix multiplication based gather instead of
61 standard tf.gather. (Default: False).
62
63 Raises:
64 ValueError: if match_results does not have rank 1 or is not an
65 integer int32 scalar tensor
66 """
67 if match_results.shape.ndims != 1:
68 raise ValueError('match_results should have rank 1')
69 if match_results.dtype != tf.int32:
70 raise ValueError('match_results should be an int32 or int64 scalar '
71 'tensor')
72 self._match_results = match_results
73 self._gather_op = tf.gather
74 if use_matmul_gather:
75 self._gather_op = ops.matmul_gather_on_zeroth_axis
76
77 @property
78 def match_results(self):
79 """The accessor for match results.
80
81 Returns:
82 the tensor which encodes the match results.
83 """
84 return self._match_results

86 def matched_column_indices(self):
87 """Returns column indices that match to some row.
88
89 The indices returned by this op are always sorted in increasing order.
90
91 Returns:
92 column_indices: int32 tensor of shape [K] with column indices.
93 """
94 return self._reshape_and_cast(tf.where(tf.greater(self._match_results, -1)))

96 def matched_column_indicator(self):
97 """Returns column indices that are matched.

99 Returns:

101 """
102 return tf.greater_equal(self._match_results, 0)
103
104 def num_matched_columns(self):
105 """Returns number (int32 scalar tensor) of matched columns."""
106 return tf.size(self.matched_column_indices())
107
108 def unmatched_column_indices(self):
109 """Returns column indices that do not match any row.
110
111 The indices returned by this op are always sorted in increasing order.
112
113 Returns:
114 column_indices: int32 tensor of shape [K] with column indices.
115 """
116 return self._reshape_and_cast(tf.where(tf.equal(self._match_results, -1)))
117
118 def unmatched_column_indicator(self):
119 """Returns column indices that are unmatched.
120
121 Returns:
122 column_indices: int32 tensor of shape [K] with column indices.
123 """
124 return tf.equal(self._match_results, -1)
125
126 def num_unmatched_columns(self):
127 """Returns number (int32 scalar tensor) of unmatched columns."""
128 return tf.size(self.unmatched_column_indices())
129
130 def ignored_column_indices(self):
131 """Returns column indices that are ignored (neither Matched nor Unmatched).
132
133 The indices returned by this op are always sorted in increasing order.
134
135 Returns:
136 column_indices: int32 tensor of shape [K] with column indices.
137 """
138 return self._reshape_and_cast(tf.where(self.ignored_column_indicator()))
139
140 def ignored_column_indicator(self):
141 """Returns boolean column indicator where True means the colum is ignored.
142
143 Returns:
144 column_indicator: boolean vector which is True for all ignored column
145 indices.
146 """
147 return tf.equal(self._match_results, -2)
148
149 def num_ignored_columns(self):
150 """Returns number (int32 scalar tensor) of matched columns."""
151 return tf.size(self.ignored_column_indices())
152
153 def unmatched_or_ignored_column_indices(self):
154 """Returns column indices that are unmatched or ignored.
155
156 The indices returned by this op are always sorted in increasing order.
157
158 Returns:
159 column_indices: int32 tensor of shape [K] with column indices.
160 """
161 return self._reshape_and_cast(tf.where(tf.greater(0, self._match_results)))
162
163 def matched_row_indices(self):
164 """Returns row indices that match some column.
165
166 The indices returned by this op are ordered so as to be in correspondence
167 with the output of matched_column_indicator(). For example if
168 self.matched_column_indicator() is [0,2], and self.matched_row_indices() is
169 [7, 3], then we know that column 0 was matched to row 7 and column 2 was
170 matched to row 3.
171
172 Returns:
173 row_indices: int32 tensor of shape [K] with row indices.
174 """
175 return self._reshape_and_cast(
176 self._gather_op(tf.cast(self._match_results, dtype=tf.float32),
177 self.matched_column_indices()))
178
179 def num_matched_rows(self):
180 """Returns number (int32 scalar tensor) of matched rows."""
181 unique_rows, _ = tf.unique(self.matched_row_indices())
182 return tf.size(unique_rows)
183
184 def _reshape_and_cast(self, t):
185 return tf.cast(tf.reshape(t, [-1]), tf.int32)

187 def gather_based_on_match(self, input_tensor, unmatched_value,
188 ignored_value):
189 """Gathers elements from `input_tensor` based on match results.
190
191 For columns that are matched to a row, gathered_tensor[col] is set to
192 input_tensor[match_results[col]]. For columns that are unmatched,
193 gathered_tensor[col] is set to unmatched_value. Finally, for columns that
194 are ignored gathered_tensor[col] is set to ignored_value.
195
196 Note that the input_tensor.shape[1:] must match with unmatched_value.shape
197 and ignored_value.shape
198
199 Args:
200 input_tensor: Tensor to gather values from.
201 unmatched_value: Constant tensor value for unmatched columns.
202 ignored_value: Constant tensor value for ignored columns.
203
204 Returns:
205 gathered_tensor: A tensor containing values gathered from input_tensor.
206 The shape of the gathered tensor is [match_results.shape[0]] +
207 input_tensor.shape[1:].
208 """
209 input_tensor = tf.concat(
210 [tf.stack([ignored_value, unmatched_value]),
211 input_tensor],
212 axis=0)
213 gather_indices = tf.maximum(self.match_results + 2, 0)
214 gathered_tensor = self._gather_op(input_tensor, gather_indices)
215 return gathered_tensor

218class Matcher(six.with_metaclass(abc.ABCMeta, object)):
219 """Abstract base class for matcher.
220 """
221
222 def __init__(self, use_matmul_gather=False):
223 """Constructs a Matcher.
224
225 Args:
226 use_matmul_gather: Force constructed match objects to use matrix
227 multiplication based gather instead of standard tf.gather.
228 (Default: False).
229 """
230 self._use_matmul_gather = use_matmul_gather
231
232 def match(self, similarity_matrix, valid_rows=None, scope=None):
233 """Computes matches among row and column indices and returns the result.
234
235 Computes matches among the row and column indices based on the similarity
236 matrix and optional arguments.
237
238 Args:
239 similarity_matrix: Float tensor of shape [N, M] with pairwise similarity
240 where higher value means more similar.
241 valid_rows: A boolean tensor of shape [N] indicating the rows that are
242 valid for matching.
243 scope: Op scope name. Defaults to 'Match' if None.
244
245 Returns:
246 A Match object with the results of matching.
247 """
248 with tf.name_scope(scope, 'Match') as scope:
249 if valid_rows is None:
250 valid_rows = tf.ones(tf.shape(similarity_matrix)[0], dtype=tf.bool)
251 return Match(self._match(similarity_matrix, valid_rows),
252 self._use_matmul_gather)
253
254 @abc.abstractmethod
255 def _match(self, similarity_matrix, valid_rows):
256 """Method to be overridden by implementations.
257
258 Args:
259 similarity_matrix: Float tensor of shape [N, M] with pairwise similarity
260 where higher value means more similar.
261 valid_rows: A boolean tensor of shape [N] indicating the rows that are
262 valid for matching.
263 Returns:
264 match_results: Integer tensor of shape [M]: match_results[i]>=0 means
265 that column i is matched to row match_results[i], match_results[i]=-1
266 means that the column is not matched. match_results[i]=-2 means that
267 the column is ignored (usually this happens when there is a very weak
268 match which one neither wants as positive nor negative example).
269 """
270 pass
