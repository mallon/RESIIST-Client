53 expected_precision = accumulated_tp_count / np.array([1, 2, 3, 4, 5, 6])
54 expected_recall = accumulated_tp_count / num_gt
55
56 precision, recall = metrics.compute_precision_recall(scores, labels, num_gt)
57 precision_float_type, recall_float_type = metrics.compute_precision_recall(
58 scores, labels_float_type, num_gt)

60 self.assertAllClose(precision, expected_precision)
