27 def test_subsample_all_examples_dynamic(self):
28 numpy_labels = np.random.permutation(300)
29 indicator = tf.constant(np.ones(300) == 1)
30 numpy_labels = (numpy_labels - 200) > 0
31
32 labels = tf.constant(numpy_labels)
33
34 sampler = (
35 balanced_positive_negative_sampler.BalancedPositiveNegativeSampler())
36 is_sampled = sampler.subsample(indicator, 64, labels)
37 with self.test_session() as sess:
38 is_sampled = sess.run(is_sampled)
39 self.assertTrue(sum(is_sampled) == 64)
