32 def _generate_label_map(self, num_classes):
33 label_map_proto = string_int_label_map_pb2.StringIntLabelMap()
34 for i in range(1, num_classes + 1):
35 item = label_map_proto.item.add()
36 item.id = i
37 item.name = 'label_' + str(i)
38 item.display_name = str(i)
39 return label_map_proto

41 def test_get_label_map_dict(self):
42 label_map_string = """
43 item {
44 id:2
45 name:'cat'
46 }
47 item {
48 id:1
49 name:'dog'
50 }
51 """
52 label_map_path = os.path.join(self.get_temp_dir(), 'label_map.pbtxt')

150 def test_keep_categories_with_unique_id(self):
151 label_map_proto = string_int_label_map_pb2.StringIntLabelMap()
152 label_map_string = """
153 item {
154 id:2
155 name:'cat'
156 }
157 item {
158 id:1
159 name:'child'
160 }
161 item {
162 id:1
163 name:'person'
164 }
165 item {
166 id:1
167 name:'n00007846'
168 }
169 """
170 text_format.Merge(label_map_string, label_map_proto)
171 categories = label_map_util.convert_label_map_to_categories(
172 label_map_proto, max_num_classes=3)
173 self.assertListEqual([{
174 'id': 2,
175 'name': u'cat'
176 }, {
177 'id': 1,
178 'name': u'child'
179 }], categories)
