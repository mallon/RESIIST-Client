36def get_image_resizer_config(model_config):

38
39 Args:

44

57def get_spatial_image_size(image_resizer_config):
58 """Returns expected spatial size of the output image from a given config.
59
60 Args:
61 image_resizer_config: An image_resizer_pb2.ImageResizer.
62
63 Returns:
64 A list of two integers of the form [height, width]. `height` and `width` are
65 set -1 if they cannot be determined during graph construction.
66
67 Raises:
68 ValueError: If the model type is not recognized.
69 """
70 if image_resizer_config.HasField("fixed_shape_resizer"):
71 return [
72 image_resizer_config.fixed_shape_resizer.height,
73 image_resizer_config.fixed_shape_resizer.width
74 ]
75 if image_resizer_config.HasField("keep_aspect_ratio_resizer"):
76 if image_resizer_config.keep_aspect_ratio_resizer.pad_to_max_dimension:
77 return [image_resizer_config.keep_aspect_ratio_resizer.max_dimension] * 2
78 else:
79 return [-1, -1]
80 if image_resizer_config.HasField(
81 "identity_resizer") or image_resizer_config.HasField(
82 "conditional_shape_resizer"):
83 return [-1, -1]
84 raise ValueError("Unknown image resizer type.")

87def get_configs_from_pipeline_file(pipeline_config_path, config_override=None):
88 """Reads config from a file containing pipeline_pb2.TrainEvalPipelineConfig.
89
90 Args:
91 pipeline_config_path: Path to pipeline_pb2.TrainEvalPipelineConfig text
92 proto.
93 config_override: A pipeline_pb2.TrainEvalPipelineConfig text proto to
94 override pipeline_config_path.
95
96 Returns:
97 Dictionary of configuration objects. Keys are `model`, `train_config`,
98 `train_input_config`, `eval_config`, `eval_input_config`. Value are the
99 corresponding config objects.
100 """
101 pipeline_config = pipeline_pb2.TrainEvalPipelineConfig()
102 with tf.gfile.GFile(pipeline_config_path, "r") as f:
103 proto_str = f.read()
104 text_format.Merge(proto_str, pipeline_config)
105 if config_override:
106 text_format.Merge(config_override, pipeline_config)
107 return create_configs_from_pipeline_proto(pipeline_config)

110def create_configs_from_pipeline_proto(pipeline_config):
111 """Creates a configs dictionary from pipeline_pb2.TrainEvalPipelineConfig.
112
113 Args:
114 pipeline_config: pipeline_pb2.TrainEvalPipelineConfig proto object.
115
116 Returns:
117 Dictionary of configuration objects. Keys are `model`, `train_config`,
118 `train_input_config`, `eval_config`, `eval_input_configs`. Value are
119 the corresponding config objects or list of config objects (only for
120 eval_input_configs).
121 """
122 configs = {}
123 configs["model"] = pipeline_config.model
124 configs["train_config"] = pipeline_config.train_config
125 configs["train_input_config"] = pipeline_config.train_input_reader
126 configs["eval_config"] = pipeline_config.eval_config
127 configs["eval_input_configs"] = pipeline_config.eval_input_reader
128 # Keeps eval_input_config only for backwards compatibility. All clients should
129 # read eval_input_configs instead.
130 if configs["eval_input_configs"]:
131 configs["eval_input_config"] = configs["eval_input_configs"][0]
132 if pipeline_config.HasField("graph_rewriter"):
133 configs["graph_rewriter_config"] = pipeline_config.graph_rewriter
134
135 return configs

153def create_pipeline_proto_from_configs(configs):
154 """Creates a pipeline_pb2.TrainEvalPipelineConfig from configs dictionary.
155
156 This function performs the inverse operation of
157 create_configs_from_pipeline_proto().
158
159 Args:
160 configs: Dictionary of configs. See get_configs_from_pipeline_file().
161
162 Returns:
163 A fully populated pipeline_pb2.TrainEvalPipelineConfig.
164 """
165 pipeline_config = pipeline_pb2.TrainEvalPipelineConfig()
166 pipeline_config.model.CopyFrom(configs["model"])
167 pipeline_config.train_config.CopyFrom(configs["train_config"])
168 pipeline_config.train_input_reader.CopyFrom(configs["train_input_config"])
169 pipeline_config.eval_config.CopyFrom(configs["eval_config"])
170 pipeline_config.eval_input_reader.extend(configs["eval_input_configs"])
171 if "graph_rewriter_config" in configs:
172 pipeline_config.graph_rewriter.CopyFrom(configs["graph_rewriter_config"])
173 return pipeline_config

176def save_pipeline_config(pipeline_config, directory):
177 """Saves a pipeline config text file to disk.
178
179 Args:
180 pipeline_config: A pipeline_pb2.TrainEvalPipelineConfig.
181 directory: The model directory into which the pipeline config file will be
182 saved.
183 """
184 if not file_io.file_exists(directory):
185 file_io.recursive_create_dir(directory)
186 pipeline_config_path = os.path.join(directory, "pipeline.config")
187 config_text = text_format.MessageToString(pipeline_config)
188 with tf.gfile.Open(pipeline_config_path, "wb") as f:
189 tf.logging.info("Writing pipeline config file to %s",
190 pipeline_config_path)
191 f.write(config_text)

194def get_configs_from_multiple_files(model_config_path="",
195 train_config_path="",
196 train_input_config_path="",
197 eval_config_path="",
198 eval_input_config_path="",
199 graph_rewriter_config_path=""):
200 """Reads training configuration from multiple config files.
201
202 Args:
203 model_config_path: Path to model_pb2.DetectionModel.
204 train_config_path: Path to train_pb2.TrainConfig.
205 train_input_config_path: Path to input_reader_pb2.InputReader.
206 eval_config_path: Path to eval_pb2.EvalConfig.
207 eval_input_config_path: Path to input_reader_pb2.InputReader.
208 graph_rewriter_config_path: Path to graph_rewriter_pb2.GraphRewriter.
209
210 Returns:
211 Dictionary of configuration objects. Keys are `model`, `train_config`,
212 `train_input_config`, `eval_config`, `eval_input_config`. Key/Values are
213 returned only for valid (non-empty) strings.
214 """
215 configs = {}
216 if model_config_path:
217 model_config = model_pb2.DetectionModel()
218 with tf.gfile.GFile(model_config_path, "r") as f:
219 text_format.Merge(f.read(), model_config)
220 configs["model"] = model_config
221
222 if train_config_path:
223 train_config = train_pb2.TrainConfig()
224 with tf.gfile.GFile(train_config_path, "r") as f:
225 text_format.Merge(f.read(), train_config)
226 configs["train_config"] = train_config
227
228 if train_input_config_path:
229 train_input_config = input_reader_pb2.InputReader()
230 with tf.gfile.GFile(train_input_config_path, "r") as f:
231 text_format.Merge(f.read(), train_input_config)
232 configs["train_input_config"] = train_input_config
233
234 if eval_config_path:
235 eval_config = eval_pb2.EvalConfig()
236 with tf.gfile.GFile(eval_config_path, "r") as f:
237 text_format.Merge(f.read(), eval_config)
238 configs["eval_config"] = eval_config
239
240 if eval_input_config_path:
241 eval_input_config = input_reader_pb2.InputReader()
242 with tf.gfile.GFile(eval_input_config_path, "r") as f:
243 text_format.Merge(f.read(), eval_input_config)
244 configs["eval_input_configs"] = [eval_input_config]
245
246 if graph_rewriter_config_path:
247 configs["graph_rewriter_config"] = get_graph_rewriter_config_from_file(
248 graph_rewriter_config_path)
249
250 return configs

311def _check_and_convert_legacy_input_config_key(key):
312 """Checks key and converts legacy input config update to specific update.
313
314 Args:
315 key: string indicates the target of update operation.
316
317 Returns:
318 is_valid_input_config_key: A boolean indicating whether the input key is to
319 update input config(s).
320 key_name: 'eval_input_configs' or 'train_input_config' string if
321 is_valid_input_config_key is true. None if is_valid_input_config_key is
322 false.
323 input_name: always returns None since legacy input config key never
324 specifies the target input config. Keeping this output only to match the
325 output form defined for input config update.
326 field_name: the field name in input config. `key` itself if
327 is_valid_input_config_key is false.
328 """
329 key_name = None
330 input_name = None
331 field_name = key
332 is_valid_input_config_key = True
333 if field_name == "train_shuffle":
334 key_name = "train_input_config"
335 field_name = "shuffle"
336 elif field_name == "eval_shuffle":
337 key_name = "eval_input_configs"
338 field_name = "shuffle"
339 elif field_name == "train_input_path":
340 key_name = "train_input_config"
341 field_name = "input_path"
342 elif field_name == "eval_input_path":
343 key_name = "eval_input_configs"
344 field_name = "input_path"
345 elif field_name == "append_train_input_path":
346 key_name = "train_input_config"
347 field_name = "input_path"
348 elif field_name == "append_eval_input_path":
349 key_name = "eval_input_configs"
350 field_name = "input_path"
351 else:
352 is_valid_input_config_key = False
353
354 return is_valid_input_config_key, key_name, input_name, field_name

357def check_and_parse_input_config_key(configs, key):
358 """Checks key and returns specific fields if key is valid input config update.
359
360 Args:
361 configs: Dictionary of configuration objects. See outputs from
362 get_configs_from_pipeline_file() or get_configs_from_multiple_files().
363 key: string indicates the target of update operation.
364
365 Returns:
366 is_valid_input_config_key: A boolean indicate whether the input key is to
367 update input config(s).
368 key_name: 'eval_input_configs' or 'train_input_config' string if
369 is_valid_input_config_key is true. None if is_valid_input_config_key is
370 false.
371 input_name: the name of the input config to be updated. None if
372 is_valid_input_config_key is false.
373 field_name: the field name in input config. `key` itself if
374 is_valid_input_config_key is false.
375
376 Raises:
377 ValueError: when the input key format doesn't match any known formats.
378 ValueError: if key_name doesn't match 'eval_input_configs' or
379 'train_input_config'.
380 ValueError: if input_name doesn't match any name in train or eval input
381 configs.
382 ValueError: if field_name doesn't match any supported fields.
383 """
384 key_name = None
385 input_name = None
386 field_name = None
387 fields = key.split(":")
388 if len(fields) == 1:
389 field_name = key
390 return _check_and_convert_legacy_input_config_key(key)
391 elif len(fields) == 3:
392 key_name = fields[0]
393 input_name = fields[1]
394 field_name = fields[2]
395 else:
396 raise ValueError("Invalid key format when overriding configs.")
397
398 # Checks if key_name is valid for specific update.
399 if key_name not in ["eval_input_configs", "train_input_config"]:
400 raise ValueError("Invalid key_name when overriding input config.")
401
402 # Checks if input_name is valid for specific update. For train input config it
403 # should match configs[key_name].name, for eval input configs it should match
404 # the name field of one of the eval_input_configs.
405 if isinstance(configs[key_name], input_reader_pb2.InputReader):
406 is_valid_input_name = configs[key_name].name == input_name
407 else:
408 is_valid_input_name = input_name in [
409 eval_input_config.name for eval_input_config in configs[key_name]
410 ]
411 if not is_valid_input_name:
412 raise ValueError("Invalid input_name when overriding input config.")
413
414 # Checks if field_name is valid for specific update.
415 if field_name not in [
416 "input_path", "label_map_path", "shuffle", "mask_type",
417 "sample_1_of_n_examples"
418 ]:
419 raise ValueError("Invalid field_name when overriding input config.")
420
421 return True, key_name, input_name, field_name

424def merge_external_params_with_configs(configs, hparams=None, kwargs_dict=None):
425 """Updates `configs` dictionary based on supplied parameters.
426
427 This utility is for modifying specific fields in the object detection configs.
428 Say that one would like to experiment with different learning rates, momentum
429 values, or batch sizes. Rather than creating a new config text file for each
430 experiment, one can use a single base config file, and update particular
431 values.
432
433 There are two types of field overrides:
434 1. Strategy-based overrides, which update multiple relevant configuration
435 options. For example, updating `learning_rate` will update both the warmup and
436 final learning rates.
437 In this case key can be one of the following formats:
438 1. legacy update: single string that indicates the attribute to be
439 updated. E.g. 'label_map_path', 'eval_input_path', 'shuffle'.
440 Note that when updating fields (e.g. eval_input_path, eval_shuffle) in
441 eval_input_configs, the override will only be applied when
442 eval_input_configs has exactly 1 element.
443 2. specific update: colon separated string that indicates which field in
444 which input_config to update. It should have 3 fields:
445 - key_name: Name of the input config we should update, either
446 'train_input_config' or 'eval_input_configs'
447 - input_name: a 'name' that can be used to identify elements, especially
448 when configs[key_name] is a repeated field.
449 - field_name: name of the field that you want to override.
450 For example, given configs dict as below:
451 configs = {
452 'model': {...}
453 'train_config': {...}
454 'train_input_config': {...}
455 'eval_config': {...}
456 'eval_input_configs': [{ name:"eval_coco", ...},
457 { name:"eval_voc", ... }]
458 }
459 Assume we want to update the input_path of the eval_input_config
460 whose name is 'eval_coco'. The `key` would then be:
461 'eval_input_configs:eval_coco:input_path'
462 2. Generic key/value, which update a specific parameter based on namespaced
463 configuration keys. For example,
464 `model.ssd.loss.hard_example_miner.max_negatives_per_positive` will update the
465 hard example miner configuration for an SSD model config. Generic overrides
466 are automatically detected based on the namespaced keys.
467
468 Args:
469 configs: Dictionary of configuration objects. See outputs from
470 get_configs_from_pipeline_file() or get_configs_from_multiple_files().
471 hparams: A `HParams`.
472 kwargs_dict: Extra keyword arguments that are treated the same way as
473 attribute/value pairs in `hparams`. Note that hyperparameters with the
474 same names will override keyword arguments.
475
476 Returns:
477 `configs` dictionary.
478
479 Raises:
480 ValueError: when the key string doesn't match any of its allowed formats.
481 """
482
483 if kwargs_dict is None:
484 kwargs_dict = {}
485 if hparams:
486 kwargs_dict.update(hparams.values())
487 for key, value in kwargs_dict.items():
488 tf.logging.info("Maybe overwriting %s: %s", key, value)
489 # pylint: disable=g-explicit-bool-comparison
490 if value == "" or value is None:
491 continue
492 # pylint: enable=g-explicit-bool-comparison
493 elif _maybe_update_config_with_key_value(configs, key, value):
494 continue
495 elif _is_generic_key(key):
496 _update_generic(configs, key, value)
497 else:
498 tf.logging.info("Ignoring config override key: %s", key)
499 return configs

502def _maybe_update_config_with_key_value(configs, key, value):
503 """Checks key type and updates `configs` with the key value pair accordingly.
504
505 Args:
506 configs: Dictionary of configuration objects. See outputs from
507 get_configs_from_pipeline_file() or get_configs_from_multiple_files().
508 key: String indicates the field(s) to be updated.
509 value: Value used to override existing field value.
510
511 Returns:
512 A boolean value that indicates whether the override succeeds.
513
514 Raises:
515 ValueError: when the key string doesn't match any of the formats above.
516 """
517 is_valid_input_config_key, key_name, input_name, field_name = (
518 check_and_parse_input_config_key(configs, key))
519 if is_valid_input_config_key:
520 update_input_reader_config(
521 configs,
522 key_name=key_name,
523 input_name=input_name,
524 field_name=field_name,
525 value=value)
526 elif field_name == "learning_rate":
527 _update_initial_learning_rate(configs, value)
528 elif field_name == "batch_size":
529 _update_batch_size(configs, value)
530 elif field_name == "momentum_optimizer_value":
531 _update_momentum_optimizer_value(configs, value)
532 elif field_name == "classification_localization_weight_ratio":
533 # Localization weight is fixed to 1.0.
534 _update_classification_localization_weight_ratio(configs, value)
535 elif field_name == "focal_loss_gamma":
536 _update_focal_loss_gamma(configs, value)
537 elif field_name == "focal_loss_alpha":
538 _update_focal_loss_alpha(configs, value)
539 elif field_name == "train_steps":
540 _update_train_steps(configs, value)
541 elif field_name == "label_map_path":
542 _update_label_map_path(configs, value)
543 elif field_name == "mask_type":
544 _update_mask_type(configs, value)
545 elif field_name == "sample_1_of_n_eval_examples":
546 _update_all_eval_input_configs(configs, "sample_1_of_n_examples", value)
547 elif field_name == "eval_num_epochs":
548 _update_all_eval_input_configs(configs, "num_epochs", value)
549 elif field_name == "eval_with_moving_averages":
550 _update_use_moving_averages(configs, value)
551 elif field_name == "retain_original_images_in_eval":
552 _update_retain_original_images(configs["eval_config"], value)
553 elif field_name == "use_bfloat16":
554 _update_use_bfloat16(configs, value)
555 else:
556 return False
557 return True

560def _update_tf_record_input_path(input_config, input_path):
561 """Updates input configuration to reflect a new input path.
562
563 The input_config object is updated in place, and hence not returned.
564
565 Args:
566 input_config: A input_reader_pb2.InputReader.
567 input_path: A path to data or list of paths.
568
569 Raises:
570 TypeError: if input reader type is not `tf_record_input_reader`.
571 """
572 input_reader_type = input_config.WhichOneof("input_reader")
573 if input_reader_type == "tf_record_input_reader":
574 input_config.tf_record_input_reader.ClearField("input_path")

576 input_config.tf_record_input_reader.input_path.extend(input_path)

578 input_config.tf_record_input_reader.input_path.append(input_path)
579 else:
580 raise TypeError("Input reader type must be `tf_record_input_reader`.")

583def update_input_reader_config(configs,
584 key_name=None,
585 input_name=None,
586 field_name=None,
587 value=None,
588 path_updater=_update_tf_record_input_path):
589 """Updates specified input reader config field.
590
591 Args:
592 configs: Dictionary of configuration objects. See outputs from
593 get_configs_from_pipeline_file() or get_configs_from_multiple_files().
594 key_name: Name of the input config we should update, either
595 'train_input_config' or 'eval_input_configs'
596 input_name: String name used to identify input config to update with. Should
597 be either None or value of the 'name' field in one of the input reader
598 configs.
599 field_name: Field name in input_reader_pb2.InputReader.
600 value: Value used to override existing field value.
601 path_updater: helper function used to update the input path. Only used when
602 field_name is "input_path".
603
604 Raises:
605 ValueError: when input field_name is None.
606 ValueError: when input_name is None and number of eval_input_readers does
607 not equal to 1.
608 """
609 if isinstance(configs[key_name], input_reader_pb2.InputReader):
610 # Updates singular input_config object.
611 target_input_config = configs[key_name]
612 if field_name == "input_path":
613 path_updater(input_config=target_input_config, input_path=value)
614 else:
615 setattr(target_input_config, field_name, value)
616 elif input_name is None and len(configs[key_name]) == 1:
617 # Updates first (and the only) object of input_config list.
618 target_input_config = configs[key_name][0]
619 if field_name == "input_path":
620 path_updater(input_config=target_input_config, input_path=value)
621 else:
622 setattr(target_input_config, field_name, value)
623 elif input_name is not None and len(configs[key_name]):
624 # Updates input_config whose name matches input_name.
625 update_count = 0
626 for input_config in configs[key_name]:
627 if input_config.name == input_name:
628 setattr(input_config, field_name, value)
629 update_count = update_count + 1
630 if not update_count:
631 raise ValueError(
632 "Input name {} not found when overriding.".format(input_name))
633 elif update_count > 1:
634 raise ValueError("Duplicate input name found when overriding.")
635 else:
636 key_name = "None" if key_name is None else key_name

644def _update_initial_learning_rate(configs, learning_rate):
645 """Updates `configs` to reflect the new initial learning rate.
646
647 This function updates the initial learning rate. For learning rate schedules,
648 all other defined learning rates in the pipeline config are scaled to maintain
649 their same ratio with the initial learning rate.
650 The configs dictionary is updated in place, and hence not returned.
651
652 Args:
653 configs: Dictionary of configuration objects. See outputs from
654 get_configs_from_pipeline_file() or get_configs_from_multiple_files().
655 learning_rate: Initial learning rate for optimizer.
656
657 Raises:
658 TypeError: if optimizer type is not supported, or if learning rate type is
659 not supported.
660 """
661
662 optimizer_type = get_optimizer_type(configs["train_config"])
663 if optimizer_type == "rms_prop_optimizer":
664 optimizer_config = configs["train_config"].optimizer.rms_prop_optimizer
665 elif optimizer_type == "momentum_optimizer":
666 optimizer_config = configs["train_config"].optimizer.momentum_optimizer
667 elif optimizer_type == "adam_optimizer":
668 optimizer_config = configs["train_config"].optimizer.adam_optimizer
669 else:
670 raise TypeError("Optimizer %s is not supported." % optimizer_type)
671
672 learning_rate_type = get_learning_rate_type(optimizer_config)
673 if learning_rate_type == "constant_learning_rate":
674 constant_lr = optimizer_config.learning_rate.constant_learning_rate
675 constant_lr.learning_rate = learning_rate
676 elif learning_rate_type == "exponential_decay_learning_rate":
677 exponential_lr = (
678 optimizer_config.learning_rate.exponential_decay_learning_rate)
679 exponential_lr.initial_learning_rate = learning_rate
680 elif learning_rate_type == "manual_step_learning_rate":
681 manual_lr = optimizer_config.learning_rate.manual_step_learning_rate
682 original_learning_rate = manual_lr.initial_learning_rate
683 learning_rate_scaling = float(learning_rate) / original_learning_rate
684 manual_lr.initial_learning_rate = learning_rate
685 for schedule in manual_lr.schedule:
686 schedule.learning_rate *= learning_rate_scaling
687 elif learning_rate_type == "cosine_decay_learning_rate":
688 cosine_lr = optimizer_config.learning_rate.cosine_decay_learning_rate
689 learning_rate_base = cosine_lr.learning_rate_base
690 warmup_learning_rate = cosine_lr.warmup_learning_rate
691 warmup_scale_factor = warmup_learning_rate / learning_rate_base
692 cosine_lr.learning_rate_base = learning_rate
693 cosine_lr.warmup_learning_rate = warmup_scale_factor * learning_rate
694 else:
695 raise TypeError("Learning rate %s is not supported." % learning_rate_type)

717def _update_generic(configs, key, value):
718 """Update a pipeline configuration parameter based on a generic key/value.
719
720 Args:
721 configs: Dictionary of pipeline configuration protos.
722 key: A string key, dot-delimited to represent the argument key.
723 e.g. "model.ssd.train_config.batch_size"
724 value: A value to set the argument to. The type of the value must match the
725 type for the protocol buffer. Note that setting the wrong type will
726 result in a TypeError.
727 e.g. 42
728
729 Raises:
730 ValueError if the message key does not match the existing proto fields.
731 TypeError the value type doesn't match the protobuf field type.
732 """
733 fields = key.split(".")
734 first_field = fields.pop(0)
735 last_field = fields.pop()
736 message = configs[first_field]
737 for field in fields:
738 _validate_message_has_field(message, field)
739 message = getattr(message, field)
740 _validate_message_has_field(message, last_field)
741 setattr(message, last_field, value)

772def _update_classification_localization_weight_ratio(configs, ratio):
773 """Updates the classification/localization weight loss ratio.
774
775 Detection models usually define a loss weight for both classification and
776 objectness. This function updates the weights such that the ratio between
777 classification weight to localization weight is the ratio provided.
778 Arbitrarily, localization weight is set to 1.0.
779
780 Note that in the case of Faster R-CNN, this same ratio is applied to the first
781 stage objectness loss weight relative to localization loss weight.
782
783 The configs dictionary is updated in place, and hence not returned.
784
785 Args:
786 configs: Dictionary of configuration objects. See outputs from
787 get_configs_from_pipeline_file() or get_configs_from_multiple_files().
788 ratio: Desired ratio of classification (and/or objectness) loss weight to
789 localization loss weight.
790 """
791 meta_architecture = configs["model"].WhichOneof("model")
792 if meta_architecture == "faster_rcnn":
793 model = configs["model"].faster_rcnn
794 model.first_stage_localization_loss_weight = 1.0
795 model.first_stage_objectness_loss_weight = ratio
796 model.second_stage_localization_loss_weight = 1.0
797 model.second_stage_classification_loss_weight = ratio
798 if meta_architecture == "ssd":
799 model = configs["model"].ssd
800 model.loss.localization_weight = 1.0
801 model.loss.classification_weight = ratio
