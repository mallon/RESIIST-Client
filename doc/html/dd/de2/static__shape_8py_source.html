26def get_dim_as_int(dim):
27 """Utility to get v1 or v2 TensorShape dim as an int.
28
29 Args:
30 dim: The TensorShape dimension to get as an int
31
32 Returns:
33 None or an int.
34 """
35 try:
36 return dim.value
37 except AttributeError:
38 return dim
