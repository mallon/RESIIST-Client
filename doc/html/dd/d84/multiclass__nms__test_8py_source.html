69 scores = np.array([[.9, 0.01], [.75, 0.05],
70 [.6, 0.01], [.95, 0],
71 [.5, 0.01], [.3, 0.01],
72 [.01, .85], [.01, .5]], np.float32)
73 score_thresh = 0.1
74 iou_thresh = .5
75 max_size_per_class = 4
76 max_output_size = 5

78 exp_nms_corners = [[0, 10, 1, 11],
79 [0, 0, 1, 1],
80 [0, 1000, 1, 1002],

83 exp_nms_classes = [0, 0, 1, 0]

85 def graph_fn(boxes, scores):
86 nms, num_valid_nms_boxes = post_processing.multiclass_non_max_suppression(
87 boxes,
88 scores,
89 score_thresh,
90 iou_thresh,
91 max_size_per_class,
92 max_total_size=max_output_size,
93 pad_to_max_output_size=True)
94 return [nms.get(), nms.get_field(fields.BoxListFields.scores),
95 nms.get_field(fields.BoxListFields.classes), num_valid_nms_boxes]
96
97 [nms_corners_output, nms_scores_output, nms_classes_output,
98 num_valid_nms_boxes] = self.execute(graph_fn, [boxes, scores])
99
100 self.assertEqual(num_valid_nms_boxes, 4)
101 self.assertAllClose(nms_corners_output[0:num_valid_nms_boxes],
102 exp_nms_corners)
103 self.assertAllClose(nms_scores_output[0:num_valid_nms_boxes],
104 exp_nms_scores)
105 self.assertAllClose(nms_classes_output[0:num_valid_nms_boxes],
106 exp_nms_classes)
