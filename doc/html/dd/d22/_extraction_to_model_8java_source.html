314 private void saveModelToXMI()
315 {
316 String fileName = XMI_FILE;
317
318 ResourceSet resSet = new ResourceSetImpl();
319 resSet.getResourceFactoryRegistry()
320 .getExtensionToFactoryMap()
321 .put("xmi", new XMIResourceFactoryImpl());
322 URI uri = URI.createURI(fileName);
323 Resource resource = resSet.createResource(uri);
324 resource.getContents().add(model);
325
326 Map<String, Object> options = new HashMap<String, Object>();
327 options.put(XMLResource.OPTION_SCHEMA_LOCATION, Boolean.TRUE);
328 options.put(XMLResource.OPTION_ENCODING, "UTF-8");
329 options.put(XMIResource.OPTION_PROCESS_DANGLING_HREF,
330 XMIResource.OPTION_PROCESS_DANGLING_HREF_DISCARD);
331 options.put(XMIResource.OPTION_USE_XMI_TYPE, true);
332
333 try {
334 resource.save(options);
335 LOGGER.log(Level.INFO, "***Model correctly generated***");
336 }
337 catch (IOException e) {
338 LOGGER.log(Level.SEVERE, "***XMI file saving problem !***");
339 }
340 }
